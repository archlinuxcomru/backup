slideshow () {
  WALLPAPERS="~/wallpapers"
  ALIST=( `ls -w1 ~/wallpapers` )
  RANGE=${#ALIST[@]}
  let "number = $RANDOM"
  let LASTNUM="`cat $WALLPAPERS/.last` + $number"
  let "number = $LASTNUM % $RANGE"
  echo $number > $WALLPAPERS/.last
  feh --bg-scale $WALLPAPERS/${ALIST[$number]}
  
  sleep 600
  slideshow
}
