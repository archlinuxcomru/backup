И прежде чем запускать всякое , отладьте иксы.
sudo systemctl disable lightdm
sudo pacman -Syyu
sudo pacman -S xorg-server xorg-xinit openbox xterm
#дальше от юзера
cp /etc/X11/xinit/xinitrc ~/.xinitrc
#исправьте ~/.xinitrc заменив строчки
#twm &
#xclock -geometry 50x50-1+1 &
#xterm -geometry 80x50+494+51 &
#xterm -geometry 80x20+494-0 &
#exec xterm -geometry 80x66+0+0 -name login
#на exec openbox-session
#перезагрузитесь и пробуйте запустить иксы
#от юзера

Какая может быть нвидия на virtualbox :(
Читайте внимательней https://wiki.archlinux.org/title/VirtualBox
вам надо virtualbox-guest-utils virtualbox-guest-iso нвидию удалить
Проверить что модули загружены
lsmod | grep vbox
должен быть vboxvideo
если нет загрузить модули
 sudo modprobe -a vboxguest vboxsf vboxvideo
