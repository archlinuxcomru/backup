заготовка .network системд и для dhcpcd и для статичного подключения подходит, нужно только подставить данные и раскомментировать или закомментировать строки.
[Match]

[Network]
DHCP=yes
#Address={address}/{mask}
#Gateway={gateway}
#DNS={dns}
