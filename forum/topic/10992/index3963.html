# устанавливаем нужные пакеты
pacman -S zsh grml-zsh-config zsh-syntax-highlighting

# добавляем zsh-syntax-highlighting в конфиг zsh
echo '[ -r /usr/share/zsh/plugins/zsh-syntax-highlighting/zsh-syntax-highlighting.zsh ] && source /usr/share/zsh/plugins/zsh-syntax-highlighting/zsh-syntax-highlighting.zsh' >> "/etc/skel/.zshrc"

# копируем .zshrc для пользователя root
cp "/etc/skel/.zshrc" "/root/.zshrc"

# Меняем оболочку по умолчанию для новых пользователей
useradd -D --shell /usr/bin/zsh
# Меняем оболочку по умолчанию для root
usermod --shell /usr/bin/zsh root
