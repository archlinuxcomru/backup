#!/usr/bin/env sh

post_install()
{
    echo OK
}

if /usr/bin/pacman ${*}
then
   post_install
fi

да, хуки пожалуй то что нужно
в /etc/pacman.d/hooks положил firefox.hook с содержимым:
[Trigger]
Operation=Install
Operation=Upgrade
Type=Package
Target=firefox

[Action]
Description=Remove getpocket
Depends=pacman
When=PostTransaction
Exec=/usr/bin/bash -c "/usr/bin/rm /usr/lib/firefox/browser/features/firefox@getpocket.com.xpi"

вроде работает
