export XLIB_SKIP_ARGB_VISUALS=1
exec vlc.real "\[email protected]"

sudo chmod 755 /usr/bin/vlc.helper
sudo mv /usr/bin/vlc{,.real}
sudo ln -sf vlc.helper /usr/bin/vlc
