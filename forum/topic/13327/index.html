Был компьютер со встроенной графикой в процессор, от него сейчас остался только винт, монитор тоже остался прежний, сменился только интерфейс подключения с DVI-D на DSUB.
Система после недолгих правок конфигов сети и загрузчика завелась, но разрешение экрана определилось как 1024x768, и поменять можно только на меньшее, а нужного 1680х1050 вообще нет.

грузился в консольном режиме, делал
Xorg :0 -configure
X -config /root/xorg.conf.new
получал полностью черный экран. При копирование конфига в /etc/X11/xorg.conf KDE вообще не запустился, пришлось удалить.
Драйвер на интегрированное Intel видео - тоже переустанавливал - результата нет.

помогите пожалуйста починить это досадное недоразумение.

Что показывает xrandr в иксах, запущенных БЕЗ xorg.conf ?
Если в списке нет нужного вам разрешения, значит либо ваш VGA-кабель (d-SUB) не поддерживает линии для передачи EDID монитора, либо сам монитор "прибедняется", и передаёт через EDID список без нужного вам режима.

Пример конфига (для nvidia) , которым я заставил работать через VGA строптивый телевизор, имеющий разрешение 1920x1080 (FullHD), но не сообщающий о поддержке этого режима через EDID.

/etc/X11/xorg.conf.d/21-screen.conf 
Section "Device"
    Identifier     "Videocard0"
    Driver         "nvidia"
    VendorName     "NVIDIA Corporation"
    Screen          0
EndSection

Section "Screen"

    Identifier     "Screen0"
    Device         "Videocard0"
    Monitor        "Monitor0"
    DefaultDepth    24
    Option         "NoLogo" "true"
    Option         "HorizSync" "CRT-0: 30-72"
    Option         "VertRefresh" "CRT-0: 56-65"
#    Option         "UseEdidFreqs" "off"
#    Option         "UseEdidDpi" "off"
    Option         "ModeValidation" "AllowNonEdidModes"
    Option         "NoTwinViewXineramaInfo" "true"
    Option         "TwinView" "0"
    Option         "DynamicTwinView" "False"
    SubSection     "Display"
        Depth       24
        Modes      "1920x1080" "1280x1024"
    EndSubSection
EndSection

Section "Monitor"
 Identifier     "Monitor0"
 HorizSync      30.0 - 73.0
 VertRefresh    56.0 - 65.0
 ModeLine       "1920x1080" 138.500 1920 1968 2000 2080 1080 1083 1088 1111 +hsync -vsync
EndSection
