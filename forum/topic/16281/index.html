как оказалось движок unity стал использовать pulseaudio по умолчанию в некоторых играх.
указание в asound.conf пульс по умолчанию решило проблему 
# Use PulseAudio by default
pcm.!default {
  type pulse
  fallback "sysdefault"
  hint {
    show on
    description "Default ALSA Output (currently PulseAudio Sound Server)"
  }
}

ctl.!default {
  type pulse
  fallback "sysdefault"
}
