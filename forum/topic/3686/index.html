  install -D -m 644 ${startdir}/src/2gis/3.0/Plugins/DGisLan/Novosibirsk.dglf "${startdir}/pkg/opt/2gis/Plugins/DGisLan/Novosibirsk.dglf" || return 1

надо заменить

install -D -m 644 ${srcdir}/2gis/3.0/Data_Novosibirsk.dgdat "${pkgdir}/opt/2gis/novosibirsk.dgdat"
  install -D -m 644 ${srcdir}/2gis/3.0/Plugins/DGisLan/Novosibirsk.dglf "${pkgdir}/opt/2gis/Plugins/DGisLan/Novosibirsk.dglf"
