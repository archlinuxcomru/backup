			.pacman-body {

				height: 4px;
				border-radius: 2px;
				position: relative;
				top: -16px;
				left: 6px;
				background: #fff;
			}
			.pacman-mouth {
				width: 0;
				height: 0;
				border-top: 4px solid transparent;
				border-right: 12px solid #fff;
				border-bottom: 4px solid transparent;
				position: relative;
				top: -16px;
				left: 8px;
			}
		</style>
	</head>
	<body>
		<div class="pacman-body"></div>
		<div class="pacman-eye"></div>
		<div class="pacman-mouth"></div>
	</body>
</html>
