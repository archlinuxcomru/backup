Нет уверенности, что понимаю вас правильно, но если мы думаем об одном и том же (цветной вывод для всего и вся?), то ставьте pacman-color и прописывайте алиасы. Например:
# If not running interactively, don't do anything
[[ $- != *i* ]] && return
if [[ $- != *i* ]] ; then
 # Shell is non-interactive.  Be done now!
 return
fi
# Aliases
alias sudo='A=`alias` sudo '
alias ls='ls --color=auto'
alias pacman='pacman-color'
