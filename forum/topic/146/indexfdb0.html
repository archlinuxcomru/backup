Как запускаете опенбокс?

Вот так запускаю себе иксы:

cp /usr/lib/systemd/system/[email protected] /etc/systemd/system/[email protected]

в /etc/systemd/system/[email protected] добавить свой логин вместо USER
ExecStart=-/sbin/agetty --noclear -a USER %I 38400

выполнить ряд команд
# systemctl daemon-reload
# systemctl disable [email protected]
# systemctl enable [email protected]
# systemctl start [email protected]

cp /etc/skel/.bash_profile ~/.bash_profile

и в этом файле добавить:
[ -z $DISPLAY && $XDG_VTNR -eq 1 ]] && exec startx

и вот мой ~/.xinitrc
#!/bin/sh
#
# ~/.xinitrc
#
# Executed by startx (run your window manager from here)
if [ -d /etc/X11/xinit/xinitrc.d ]; then
  for f in /etc/X11/xinit/xinitrc.d/*; do
    [ -x "$f" ] && . "$f"
  done
  unset f
fi
exec dbus-launch openbox-session
