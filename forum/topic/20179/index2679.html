Пожалуйста пользуйтесь поиском по форуму
https://archlinux.org.ru/forum/topic/20130/

и вики
https://wiki.archlinux.org/index.php/Conky#Config_file_syntax_changed

A Lua script is available to convert from the old syntax to the new Lua syntax here https://github.com/brndnmtthws/conky/blob/master/extras/convert.lua

Сконвектированный conkyrc
conky.config = {
	use_xft = true,
	font = '123:size=8',
	xftalpha = 0.1,
	update_interval = 1,
	total_run_times = 0,

	own_window = true,
	own_window_type = 'normal',
	own_window_transparent = true,
	own_window_hints = 'undecorated,below,sticky,skip_taskbar,skip_pager',
	own_window_colour = '#000000',
	own_window_argb_visual = true,
	own_window_argb_value = 0,

	double_buffer = true,
--minimum_size 250 5
--maximum_width 500
	draw_shades = false,
	draw_outline = false,
	draw_borders = false,
	draw_graph_borders = true,
	default_color = 'white',
	default_shade_color = 'red',
	default_outline_color = 'green',
	alignment = 'top_left',
	gap_x = 1140,
	gap_y = 320,
	no_buffers = true,
	uppercase = false,
	cpu_avg_samples = 2,
	net_avg_samples = 1,
	override_utf8_locale = true,
	use_spacer = 'yes',
	minimum_width = 0, minimum_height = 0,
};

conky.text = [[
${voffset 10}${color EAEAEA}${font GE Inspira:pixelsize=55}${time %H:%M:%S}${font}${voffset -84}${offset 10}

${voffset 50}${offset 12}${font Ubuntu:pixelsize=15}

${color FFA300}HDD ${offset 9}$color${fs_free_perc /home}%
${color FFA300}HDD ${offset 9}$color${fs_free /} / ${fs_size /}
${voffset 0}${color #ffffff}${diskiograph sda 20,200 fb5500 0eff00 scale -l}
${color FFA300}RAM ${offset 9}$color${memperc}%
${color FFA300}RAM ${offset 9}$color$mem / $memmax

${color FFA300}CPU ${offset 9}$color${cpu cpu0}%
${color FFA300}TEMP ${offset 9}$color${hwmon 0 temp 1} °C
${voffset 0}${color}${cpugraph 20,200 1B3959 DDAA00 scale}]];
