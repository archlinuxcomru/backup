GRUB_CMDLINE_LINUX_DEFAULT="loglevel=3 quiet splash cryptdevice=UUID=ххххххх-хххх-хххх-хххх-ххххххх:cryptdisk root=/dev/mapper/cryptdisk resume=/dev/mapper/swap"

HOOKS=(base udev autodetect modconf block encrypt resume filesystems keyboard fsck)
