Я полностью победил тиринг везде где можно следующими настройками....если что тоже гибридная графика Intel + Nvidia

/etc/profile.d/kwin.sh
export KWIN_TRIPLE_BUFFER=1
/etc/X11/xorg.conf.d/20-intel.conf
Section "Device"
        Identifier  "Intel Graphics"
        Driver      "intel"
        BusID       "PCI:0:2:0"
        Option      "AccelMethod" "SNA"
        Option      "TearFree" "true"
        Option      "DRI"    "3"
        Option      "TripleBuffer" "true"
EndSection
