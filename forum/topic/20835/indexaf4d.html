Как выключить DM
https://wiki.archlinux.org/title/Systemd#Using_units
https://wiki.archlinux.org/title/LightDM#Enabling_LightDM
systemctl disable lightdm
перед этим
https://wiki.archlinux.org/title/Xinit
pacman -S xorg-xinit
создать нормальный ~/.xinitrc с вашим DE
#!/bin/sh

userresources=$HOME/.Xresources
usermodmap=$HOME/.Xmodmap
sysresources=/etc/X11/xinit/.Xresources
sysmodmap=/etc/X11/xinit/.Xmodmap

# merge in defaults and keymaps

if [ -f $sysresources ]; then
    xrdb -merge $sysresources

fi

if [ -f $sysmodmap ]; then
    xmodmap $sysmodmap
fi

if [ -f "$userresources" ]; then
    xrdb -merge "$userresources"

fi

if [ -f "$usermodmap" ]; then
    xmodmap "$usermodmap"
fi
#
# start some nice programs

if [ -d /etc/X11/xinit/xinitrc.d ] ; then
 for f in /etc/X11/xinit/xinitrc.d/?*.sh ; do
  [ -x "$f" ] && . "$f"
 done
 unset f
fi
#exec i3
#exec startplasma-x11
#exec startlxde
#exec io.elementary.cerbere
# startdde
#exec startxfce4
exec cinnamon-session
вместо cinnamon-session ваше DE если не cinnamon
