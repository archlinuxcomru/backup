Кусок моего ~./.fluxbox/menu
...
[submenu] (Инструменты)
      [exec] (Имя окна)			 {xprop WM_CLASS|cut -d \" -f 2|xmessage -file - -center} 
      [exec] (Снимок экрана - JPG)	 {import screenshot.jpg && display -resize 50% screenshot.jpg} 
      [exec] (Снимок экрана - PNG)	 {import screenshot.png && display -resize 50% screenshot.png} 
      [exec] (Выполнить)		 {fbrun -font 10}
