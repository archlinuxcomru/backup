Вот сам PKGBUILD…

pkgname=lotus-notes-base
pkgver=8.5
pkgrel=1
_pkgnum=C1SC0EN
_pkgorig=ibm_lotus_notes
pkgdesc='Lotus Notes messaging and collaboration suite'
arch=(i686)
license=('unknown')
url='http://www.ibm.com/software/lotus/products/notes/'
install=${pkgname}.install
depends=('alsa-lib' 'atk' 'audiofile' 'avahi' 'cairo' 'libstdc++5' 'libcups' 'dbus' 'dbus-glib' 'e2fsprogs' 'eel' 'esound' 'expat' 'fontcon$
makedepends=('rpmextract')
source=(${_pkgnum}.tar
        ${pkgname}.install)
md5sums=('3764cc65d2224ab6098c6ac02b40640c'
        '19a68a6258071d58b2f14cc344a76d8a'
        'd44beff80d625ad072a47368e12068ff')
# This package must first be downloaded from IBM's Passport Advantage website:
# https://www-112.ibm.com/software/howtobuy/passportadvantage/paocustomer/passporthome.nsf/pao?OpenAgent&softdlmediaaccess
build() {
        # Unpack source packages
        cd ${pkgdir}
        rpmextract.sh ../src/${_pkgorig}-${pkgver}.i586.rpm
        # fix ownership and permissions
        chown root:root -R ${pkgdir}
}

Только сомневаюсь, что без ручного вмешательства у Вас все заработает. Ибо пакет в AUR не обновлялся с 2009 года, а родной IBM-овский deb-пакет уже тогда требовал допила даже на ubuntu. Проблема в том, что библиотеки gtk уже давно везде обновились, а в инсталляшке от IBM требуются старые.
