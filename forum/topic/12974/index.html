о, как можно)))
#!/bin/bash
bri_in=$(cat /sys/class/backlight/intel_backlight/brightness)
bri_down ()
{
if (( "$bri_in" >= "56" ));
        then
                let bri=$bri_in-50;
                echo $bri > /sys/class/backlight/intel_backlight/brightness;
fi;
}
bri_up ()
{
if (( "$bri_in" <= "925" ));
        then
                let bri=$bri_in+50;
                echo $bri > /sys/class/backlight/intel_backlight/brightness;
fi;
}
while getopts ":du" option ;
do
    case $option in
        d) bri_down
        ;;
        u) bri_up
        ;;
        *) echo "error";
            exit 1
        ;;
        esac
done
