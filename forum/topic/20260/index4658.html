Давайте пока разберемся с пульсой , потом уже с polybar
sudo pacman -Syu  pulseaudio pulseaudio-alsa pavucontrol
systemctl --user stop pulseaudio.socket
systemctl --user stop pulseaudio.service
rm -rf  ~/.pulse ~/.pulse-cookie ~/.config/pulse ~/.config/autostart/pulseaudio.desktop
sudo rm /etc/xdg/autostart/pulseaudio.desktop

systemctl reboot
после перезагрузки покажите вывод команд

ps aux | grep -i pulse
 env |grep -i -e user -e home
 systemctl --user status  pulseaudio.service
journalctl -b | grep pulse
pavucontrol
