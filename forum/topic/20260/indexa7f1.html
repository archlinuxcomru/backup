Как убрать не нужные rEFIndы?
смотрим номер записей
 efibootmgr -v
Boot0001 запись номер один например
удаляем
sudo efibootmgr -Bb номер ненужной записи
# например
sudo efibootmgr -Bb 0003
sudo efibootmgr -Bb 0002

если надо порядок изменить (BootOrder: 0003,0001,0004,0005,0002)
sudo efibootmgr -o все записи в нужном порядке через запитую
# например
sudo efibootmgr -o 0001,0003,0002,0004,0005

Сейчас их там штук 5: винда, линукс, хром, сентос, арч. Как это сделать? Закоментировать в конфиге не нужные мне menuentry?
закомментировать ненужные
и отключить автосканирование на том диске или дисках где это не надо
например
uefi_deep_legacy_scan false
dont_scan_volumes "ARCH1" "EFI system partition" "EFI_ARCH" "arch" "M-PM-^RM-PM->M-QM-^AM-QM-^AM-QM-^BM-PM-0M-PM-=M-PM->M-PM-2M-PM-8M-QM-^BM-QM-^L" "vindata"
dont_scan_files shim.efi,MokManager.efi
