Если опять сглючит , то можно для проверки на флешку рефинд поставить
создать на флешке fat32 раздел и поставить на него рефинд
 sudo refind-install  --usedefault  /dev/раздел fat32 флешки
потом в настройки биоса войти и поставить загрузку с флешки
systemctl reboot --firmware-setup
или в меню систембут выбрать ребут в фирмаре интерфейс

Почему то не отображается у вас без нажатия стрелочки экран, проверить глюк systemboot или что другое

Ненужные записи можно удалить
sudo efibootmgr -Bb номер ненужной записи
sudo efibootmgr -Bb 0003
sudo efibootmgr -Bb 0002
но это уже после рабочей флешки, efibootmgr почему то глючит у вас

ну значит системдбут не показывает из за своих дров на видео
можно рефинд использовать тогда
Подправьте конфиг /boot/EFI/Boot/refind.conf
добавив в конец
menuentry "Arch2" {
    icon    /EFI/Boot/icons/os_chrome.png
    loader   /vmlinuz-linux
    initrd   /initramfs-linux-fallback.img
    options  "root=UUID=b43f80fc-9efa-464c-a9c0-2dbf3bdbafe7 rootfstype=ext4 rootflags=rw  add_efi_memmap  initrd=\intel-ucode.img"
}
menuentry "Systemdboot" {
icon      /EFI/Boot/icons/os_centos.png
loader  \EFI\systemd\systemd-bootx64.efi
}
