Есть ли тема или способ вывод ОС в столбик, а не в строчку (не помещаются все, если еще и флешки подключить)
Вообще-то у большинства экранов ширина больше чем высота.
В строчку больше помещается чем в столбик (если конечно не в текстовом режиме грузитесь)

Чтобы не мешали лишние пункты их показ можно отключить

пример части refind.conf показываются только две иконки арча и винды и кнопки выключить перезагрузить
...
use_graphics_for linux
showtools shutdown,reboot
hideui label
hideui badges
hideui hints
uefi_deep_legacy_scan false
dont_scan_volumes "ARCH1" "EFI system partition" "EFI_ARCH" "arch"  "vindata"
dont_scan_files shim.efi,MokManager.efi
scan_all_linux_kernels false
extra_kernel_version_strings linux-lts,linux
menuentry "Arch Linux" {
    icon /EFI/Boot/icons/os_arch.png
    volume   ARCH1
    loader   /boot/vmlinuz-linux
    initrd   /boot/initramfs-linux.img
    options  "root=UUID=f380ec9b-ada5-4f62-a9be-588ef0c955f5 rootfstype=ext4 module_blacklist=8021q,cfg80211,pcspkr rw add_efi_memmap"
    submenuentry "Boot using fallback initramfs" {
        initrd /boot/initramfs-linux-fallback.img
    }
    submenuentry "Boot to terminal" {
        add_options "systemd.unit=multi-user.target"
    }

}

menuentry "Windows" {
    icon /EFI/Boot/icons/os_win8.png
    loader \EFI\Microsoft\Boot\bootmgfw.efi

}

...
Если при этом иногда надо автосканирование загрузчиков можно просто добавить отдельный пункт в меню с загрузкой рефинд с отдельной папки с другим конфигом
примерно
menuentry "Refind Skan" {
    icon /EFI/Boot/icons/нужная_иконка.png
    loader \EFI\RefindSkan\refind_x64.efi

}
При этом еще уменьшить размер иконок
