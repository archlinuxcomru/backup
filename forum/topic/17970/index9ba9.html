# echo -n 8 | dd of=/usr/bin/skype bs=1 seek=23370360 conv=notrunc
Если комбинация 4.3.0.37 встречается в файле только один раз и начинается с адреса
23370361 (1649A79), то это равносильно патчу
sed -i 's/4\.3\.0\.37/8\.3\.0\.37/'
Может кто вывести
# hexdump -C -s 23370360 -n 8 /usr/bin/skype
