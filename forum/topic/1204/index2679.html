Вот тебе для примера мой
$ cat .xinitrc 
DEFAULT_SESSION=openbox
case $1 in
openbox)
    exec dbus-launch --sh-syntax --exit-with-session openbox-session
    ;;
kde4)
    exec startkde
	;;
wmfs)
    exec wmfs
    ;;
dwm)
    feh --bg-scale /home/alx/wall/vladstudio_quiet_night_1920x1080_my.jpg &
    conky -c ~/.conkyrc_dwm | while read -r; do xsetroot -name "$REPLY"; done &
    yeahconsole &
    exec dbus-launch --sh-syntax --exit-with-session dwm
    ;;
*)
    exec dbus-launch --sh-syntax --exit-with-session $DEFAULT_SESSION
    ;;
esac
Ну и в /etc/slim.conf
# Available sessions (first one is the default).
# The current chosen session name is replaced in the login_cmd
# above, so your login command can handle different sessions.
# see the xinitrc.sample file shipped with slim sources
sessions            openbox,kde4,wmfs,dwm
