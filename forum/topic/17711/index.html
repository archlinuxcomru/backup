На примере WarThunder
Команда запуска
wr
#!/bin/bash
xinit '/home/oleg/bin/wart.st' -- :2

скрипт wart.st
#!/bin/bash
#переключаем на русскую раскладку
setxkbmap -layout 'ru'
#устанавливаем чувствительность и ускорение мыши
xinput set-prop 8 280 -1
xinput --set-prop 8 "Device Accel Constant Deceleration" 2.0
#устанавливаем громкость
amixer set Master 210
#переходим в каталог с игрой
cd /home/oleg/WarThunder
#запускаем старт игры
/home/oleg/WarThunder/linux64/aces

Для запуска иксов от юзера
создать /etc/X11/Xwrapper.config
allowed_users=anybody
needs_root_rights = yes
