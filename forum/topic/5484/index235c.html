/etc/X11/xorg.conf.d/keyboard.conf 
Section "InputClass"
	Identifier	"Keyboard Defaults"
	MatchIsKeyboard	"yes"
	Option		"pc104"
	Option		"XkbLayout"	"us,ru"
	Option		"XkbOptions"	"grp:alt_shift_toggle,grp_led:scroll"
EndSection
