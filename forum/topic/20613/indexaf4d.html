Спасибо, сильный довод, поправил подход -


cat /usr/share/thumbnailers/pec.thumbnailer

[Thumbnailer Entry]
TryExec=/usr/bin/inkscape
Exec=inkscape  --export-type=png --export-dpi=30  %i --export-filename=%o
MimeType=application/x-pecfile;

cat /usr/share/mime/packages/pec-mime.xml

<?xml version='1.0' encoding='utf-8'?>
<mime-info xmlns="http://www.freedesktop.org/standards/shared-mime-info">
  <mime-type type="application/x-pecfile">
    <comment>Brother PEC file</comment>
    <glob pattern="*.pec"/>
    <glob pattern="*.PEC"/>
  </mime-type>
</mime-info>

update-mime-database /usr/share/mime

   width="210mm">
     inkscape:document-units="mm"

cat '/home/oleg/Рабочий стол/accessories-text-editor-symbolic.svg'    |grep -e document-units -e width -e  height
<svg xmlns="http://www.w3.org/2000/svg" width="16" height="16.007".....
