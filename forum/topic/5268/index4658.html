Ну и стрелку в главном меню тогда уж убрать бы. Что-то типа такого, если не ошибаюсь:
# The following removes the arrows from the panel
style "panel-arrow-remove"
{
engine "pixmap"
	{
	image
	{
		function	= ARROW
		recolorable	= TRUE
		overlay_file	= "Panel/arrow-blank.png"
		overlay_border	= {2,2,2,2}
		overlay_stretch	= FALSE
		arrow_direction	= UP
	}
	image
	{
		function	= ARROW
		recolorable	= TRUE
		overlay_file	= "Panel/arrow-blank.png"
		overlay_border	= {2,2,2,2}
		overlay_stretch	= FALSE
		arrow_direction	= DOWN
	}
	}
}
widget_class "*PanelToplevel*" 			style "panel-arrow-remove"
