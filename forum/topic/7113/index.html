abs extra/firefox
cd /var/abs/extra/firefox
sed -i "s/$(md5sum mozconfig | cut -d" " -f1; sed -i 's/ac_add_options --with-branding=browser\/branding\/unofficial/ac_add_options --enable-official-branding/' mozconfig)/$(md5sum mozconfig | cut -d" " -f1)/" PKGBUILD
