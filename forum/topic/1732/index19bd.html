Всё, разобрался. Добавил в ~/.xinitrc 
if [ -d /etc/X11/xinit/xinitrc.d ]; then
  for f in /etc/X11/xinit/xinitrc.d/*; do
    [ -x "$f" ] && . "$f"
  done
  unset f
fi
из /etc/skel/.xinitrc. Удалил юзера из всех групп кроме wheel и vboxusers. И всё путём.
