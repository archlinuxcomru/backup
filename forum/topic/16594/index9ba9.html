Возвращаясь к теме. У меня неактивные окна слегка притемняются средствами compton. На гномоокна это тоже действует. Тени от окон - штука необязательная, то тоже улучшает восприятие.
Вот часть ~/.config/compton.conf:
inactive-dim = 0.4; # Dim inactive windows. (0.0 - 1.0, defaults to 0).
focus-exclude = [
  "name = 'vdpau'",
  "name = 'MPlayer'",
  "name = 'conky_clock'",
  "class_g = 'vdpau'",
  "class_g = 'MPlayer'",
  "class_g = 'Conky'"
  ]
# Shadow
shadow = true;			# Enabled client-side shadows on windows.
no-dock-shadow = true;		# Avoid drawing shadows on dock/panel windows.
no-dnd-shadow = true;		# Don't draw shadows on DND windows.
clear-shadow = true;		# Zero the part of the shadow's mask behind the window (experimental).
shadow-radius = 7;		# The blur radius for shadows. (default 12)
shadow-offset-x = -7;		# The left offset for shadows. (default -15)
shadow-offset-y = -7;		# The top offset for shadows. (default -15)
shadow-ignore-shaped = true;
shadow-exclude = [
  "name = 'Notification'",
  "override_redirect",
  "class_g = 'Conky'",
  "class_g ?= 'Notify-osd'",
  "_GTK_FRAME_EXTENTS@:c"
  ];
Наверно, в focus-exclude половина лишнего, но неохота разбираться.
