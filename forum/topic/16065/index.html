#!/bin/bash
_stat(){

  cat /proc/`pidof $1`/status | grep 'State:' | awk '{print $2}'
fi
}

while true; do
  if [[ $(_stat plugin-container) == 'R' || $(_stat firefox) == 'R' ]];then
    xset s reset
  fi
