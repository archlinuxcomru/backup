Вообщем автоматизировал настройки вот так:
#!/bin/sh
clear # Очистка экрана
echo Введите путь:
read B
echo "Вы ввели : $B"
#sleep 1s
cd $B
ls -F | grep / | sed 's/\///g'
echo Введите путь:
read L
mkdir $L
cd $L
pwd
while :
do
echo "Выберите интересующую Вас пункт:"
echo
echo "[A]Pon mediana"
echo "[B]Pptpsetup"
echo "[C]Pppd"
echo "[Z]Exit"
echo
read person
case "$person" in
# Обратите внимание: переменная взята в кавычки.
  "A" | "a" )
  # Пользователь может ввести как заглавную, так и строчную букву.
  A="vpn_pon.log"
  echo $A
  pwd
 ...
#echo Ввожу команду: nano /etc/ppp/options.ppp0 >> $A
#echo "#Коментарий" > /etc/ppp/options.ppp0
#nano /etc/ppp/options.ppp0
#cat /etc/ppp/options.ppp0 >> $A
  echo Ввожу команду: pon mediana >> $A
  pon mediana >> $A  3>&2  2>&1
#echo Ввожу команду: pppd call middinna nodetach debug >> $A
#pppd call middinna nodetach debug >> $A  3>&2  2>&1
  echo Ввожу команду: route add default dev ppp0 >> $A
  route add default dev ppp0 >> $A  3>&2  2>&1
  echo Ввожу команду: ping -c2 ya.ru >> $A
  ping -c2 ya.ru >> $A  3>&2  2>&1
  echo "The End"
  ;;
# Обратите внимание: блок кода, анализирующий конкретный выбор, завершается
# двумя символами "точка-с-запятой".
  "B" | "b" )
  echo
  A="vpn_pptp.log"
  echo $A
  pwd
   ;;
  "C" | "c" )
  echo
  A="vpn_pppd.log"
  echo $A
  pwd
  echo "Тут будет еще что-то" > $A
  echo "The End" >> $A
    ;;
  "Z" | "z" )
  exit 0
  ;;
          * )
   # Выбор по-умолчанию.
   # "Пустой" ввод тоже обрабатывается здесь.
   echo
   echo "Нет данных."
  ;;
esac
echo "------------"
done
#exit 0
