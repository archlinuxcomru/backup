gdm надо выключать и отлаживать просто иксы если вайланд не используете
sudo pacman -S xorg-xinit
#от юзера
cp /etc/X11/xinit/xinitrc ~/.xinitrc
nano ~/.xinitrc
#убрать или закомментить строчки
# #twm &
# #xclock -geometry 50x50-1+1 &
# #xterm -geometry 80x50+494+51 &
# #xterm -geometry 80x20+494-0 &
# #exec xterm -geometry 80x66+0+0 -name login
#добавить exec нужная сессия, например
##exec i3
##exec startplasma-x11
##exec startlxde
##exec io.elementary.cerbere
## startdde
##exec startxfce4
#отключить gdm
systemctl disable gdm.service
#перезагрузиться и пробовать запускать иксы смотреть лог иксов
startx
nano /var/log/Xorg.0.log
