Я конечно не гуру пакетособиратель)) и меня ща запинают, но ради прикола решил попробывать. Вот чего наклепал


PKBUILD
# Id: The DefCon Epplet build on 2000/11/20
# Maintainer:
# Contributor: none
pkgname=DefCon
pkgver=0.4
pkgrel=1
pkgdesc="Enlightnment epplet DefCon - shows system load"
arch=(i686 x86_64)
url="http://tr1tium.com/projects/epplet/"
license="unknown"
groups=
provides=
depends=('e-svn' 'epplet-base>=1.0.0' 'esound')
makedepends=
conflicts=
replaces=
backup=
install=
source=(http://tr1tium.com/projects/epplet/$pkgname-$pkgver.tgz)
md5sums=('71946576a603037e0e6434f960216e67')
build() {
   cd $startdir/src/$pkgname
   make clean || return 1
   make || return 1
   make prefix=$startdir/pkg/usr install
}

Needed packages:

	epplet-base-1.0.0 (or higher)
