После того, как я первый раз по невнимательности отправил комп в ребут, откорректировал свои скрипты для выключения/перезагрузки, внеся в них поддтверждение
#!/bin/sh
zenity  --question --text "Вы действительно желаете выключить компьтер?"
if [ $? = 0 ]
	then systemctl --user stop transmission && systemctl poweroff
fi
exit 0

#!/bin/sh
zenity  --question --text "Вы действительно желаете перезагрузить компьтер?"
if [ $? = 0 ]
	then systemctl --user stop transmission && systemctl reboot
fi
exit 0
