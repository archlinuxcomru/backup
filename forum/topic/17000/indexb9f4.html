declare font='xos4 terminus:pixelsize=16:style=bold'
declare exp='onstart=uncollapse;button1=exit;button3=exit'
tm=$(date +%m)
ty=$(date +%Y)
while true; do
  var=$(cat /tmp/var)
  if [[ $var == + ]]; then
    if (( "$tm" > "11" )); then
      tm=$(expr $tm + 1 - 12)
      ty=$(expr $ty + 1)
    else
      tm=$(expr $tm + 1)
    fi
  elif [[ $var == - ]]; then
    if (( "$tm" < "2" )); then
      tm=$(expr $tm - 1 + 12)
      ty=$(expr $ty - 1)
    else
      tm=$(expr $tm - 1)
    fi
  fi
  echo "^ca(1,echo '+' > /tmp/var)+^ca()" > /tmp/dzen_cal
  echo "^ca(1,echo '-' > /tmp/var)-^ca()" >> /tmp/dzen_cal
  cal $tm $ty >> /tmp/dzen_cal
  echo '' > /tmp/var
  cat /tmp/dzen_cal
  sleep 1
done | dzen2 -p -e "$exp" -fn "$font" -w 200 -l 10 -y 26 -x -200
