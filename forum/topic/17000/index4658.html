------------------------------------------------------------------------
-- workspace --
--------------------------------------------------------------------------
myWorkspaces    :: [String]
myWorkspaces    = clickable $ [ "^i(/home/tozen/.xmonad/.icons/term.xbm)  Term "++fro++""
                              , "^i(/home/tozen/.xmonad/.icons/www.xbm)  Web "++fro++""
                              , "^i(/home/tozen/.xmonad/.icons/code.xbm)  Code "++fro++""
                              , "^i(/home/tozen/.xmonad/.icons/media.xbm)  Music "++fro++""
                              , "^i(/home/tozen/.xmonad/.icons/file2.xbm)  Files "++fro++""
                              ]
                where clickable l       = [ "^ca(1,xdotool key super+" ++ show (n) ++ ")" ++ ws ++ "^ca()" |
                                            (i,ws) <- zip [1..] l,
                                            let n = i ]
