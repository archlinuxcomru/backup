Установите caffeine-ng и когда нужно будете блокировать хранитель экрана.
Для этого не нужно ничего устанавливать, тем более из неофициальных реп.
#!/bin/sh
if [[ `xset q | grep "DPMS is Enabled"` ]]
	then xset -dpms s off
	notify-send -t 5000 -u normal "Screensaver:" "Хранитель экрана выключен"
	else xset +dpms s 300
	notify-send -t 5000 -u normal "Screensaver:" "Хранитель экрана включен"
fi
sleep 1
Биндите выполнение скрипта на желаемый хоткей.

#!/bin/bash
while true; do
  _timeout=60
  _time=`xset -q | grep 'timeout:' | awk '{print $2}'`
  if [[ $_time != 0 ]];then
    if [ -f /proc/`pgrep firefox`/status ]; then
      _stat=$(cat /proc/`pgrep firefox`/status | grep 'State:' | awk '{print $2}')
      if [[ $_stat == 'R' ]];then
        xset s off
      fi
    fi
  else
    if [[ $_time != $_timeout ]];then
      xset s $_timeout
    fi
  fi
sleep 1
done
