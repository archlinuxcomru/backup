$ cat /etc/rc.d/mpd-status
#!/bin/bash
. /etc/rc.conf
. /etc/rc.d/functions
PID=`pidof -o %PPID -x /usr/local/bin/mpd-status.sh`
case "$1" in
  start)
    stat_busy "Starting MPD-Status"
    if [ -z "$PID" ]; then
      # check dependencies:
      # 1. screen need for start script (/usr/local/bin/mpd-status.sh)
      type -P screen&>/dev/null || { stat_fail; echo "Please install screen"; exit 0; }
      # 2. mpc - for control mpd
      type -P mpc&>/dev/null || { stat_fail; echo "Please install mpc"; exit 0; }
      # start script
      su root -c 'screen -d -m /usr/local/bin/mpd-status.sh' &> /dev/null
    fi
    if [ $? -gt 0 ]; then
      stat_fail
    else
      add_daemon "mpd-status"
      stat_done
    fi
  ;;
  stop)
    stat_busy "Stopping MPD-Status"
    [ ! -z "$PID" ] && kill $PID &>/dev/null
    if [ $? -gt 0 ]; then
      stat_fail
    else
      rm_daemon "mpd-status"
      stat_done
    fi
  ;;
всё, что тебе нужно - изменить/удалить несколько строк и запихнуть это дело в /etc/rc.d/<script-name>
потом всё до безобразия просто:
# /etc/rc.d/<script-name> <start|stop>
з.ы.: если хочешь, чтобы скрипт запускался автоматически - добавь его в DAEMONS=(… <script-name> …) в /etc/rc.conf
