Доброго времени суток, собственно проблему с натяжкой можно назвать проблемой, но все же ее устранение даст весьма много позитива. Суть загвоздки в том что клавиша Fn работает наоборот. Если задействовать с ней клавиши f1-fx или другие системные, то они срабатывают как обычные. А если без фн то как системные. Т.е. ситуация просто “наоборот”. В целом оно делу не мегает, но вот при использовании двупанельного файлового менеджера сами понимаете каждый раз тыркать на фн весьма утомительно.
┌─[ Вт. авг. 30 ] :: [jinn] - [archjinn] :: [~]
└─>> cat /etc/X11/xorg.conf.d/10-evdev.conf
#
# Catch-all evdev loader for udev-based systems
# We don't simply match on any device since that also adds accelerometers
# and other devices that we don't really want to use. The list below
# matches everything but joysticks.
Section "InputClass"
        Identifier "evdev pointer catchall"
        MatchIsPointer "on"
        MatchDevicePath "/dev/input/event*"
        Driver "evdev"
EndSection
Section "InputClass"
        Identifier "evdev keyboard catchall"
        MatchIsKeyboard "on"
        MatchDevicePath "/dev/input/event*"
        Driver "evdev"
EndSection
Section "InputClass"
        Identifier "evdev touchpad catchall"
        MatchIsTouchpad "on"
        MatchDevicePath "/dev/input/event*"
        Driver "evdev"
EndSection
Section "InputClass"
        Identifier "evdev tablet catchall"
        MatchIsTablet "on"
        MatchDevicePath "/dev/input/event*"
        Driver "evdev"
EndSection
Section "InputClass"
        Identifier "evdev touchscreen catchall"
        MatchIsTouchscreen "on"
        MatchDevicePath "/dev/input/event*"
        Driver "evdev"
EndSection
-------------------------------------------------------------------
┌─[ Вт. авг. 30 ] :: [jinn] - [archjinn] :: [~]
└─>> cat /etc/X11/xorg.conf.d/20-keyboard.conf 
#Keyboard sections . User set . Alt+Shift layout switcher
Section "InputClass"
        Identifier "evdev keyboard catchall"
        MatchIsKeyboard "on"
        MatchDevicePath "/dev/input/event*"
        Driver "evdev"
	Option "XkbLayout" "us,ru(winkeys)"
	Option "XkbOptions" "grp:alt_shift_toggle"
EndSection
