root/.compiz/.....
поэтому и пишет : No such file or directory
Файлы будут в домашней директории пользователя и запускать надо из под пользователя
Вот как у меня запускается из ~/.xinitrc
#!/bin/bash

eval `dbus-launch --sh-syntax --exit-with-session`

sleep 3 && ~/.autostart.sh --exit &

exec compiz ccp
в файле ~/.autostart.sh пишу то что надо дополнительно запустить
