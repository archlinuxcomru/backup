Я L2TP/xl2tpd использовал ещё на старом rc.conf'е, как этим нынче управлять через netcfg не знаю (сменил провайдера давно), но покажи, на всякий случай, конфиги - может и придёт что-то на ум.
По идее, то, что раньше я писал непосредственно в rc.conf, можно прописать и в /etc/ppp/ip-up.d/ip-up-route.sh (создать его), типа:
#!/bin/sh
PATH="/sbin:/bin:/usr/sbin:/usr/bin"
GATEWAY=`route | grep ^default | awk '{print $2}'`
route add -host tp.corbina.net gw ШЛЮЗ dev eth0
route add -host DNS1 gw ШЛЮЗ dev eth0
route add -host DNS2 gw ШЛЮЗ dev eth0
route del default
route add default dev ppp0
(
echo '#!/bin/sh'
echo 'PATH="/sbin:/bin:/usr/sbin:/usr/bin"'
echo 'route del default'
echo "route add default gateway $GATEWAY"
) >/etc/ppp/ip-down

/etc/xl2tpd/xl2tpd.conf
[global]
access control = yes
auth file = /etc/ppp/chap-secrets
[lac corbina]
lns = tp.internet.beeline.ru
redial = yes
redial timeout = 5
require chap = yes
require authentication = no
name = 1234567890
ppp debug = no
pppoptfile = /etc/ppp/options.xl2tpd
require pap = no
autodial = yes
/etc/ppp/options.l2tpd
lcp-echo-interval 10
lcp-echo-failure 2
name 1234567890
remotename tp
ipparam corbina
connect /bin/true
mru 1460
mtu 1460
nodeflate
noauth
persist
maxfail 0
nobsdcomp
noaccomp
nopcomp
defaultroute
/etc/rc.conf (описание сетевых интерфейсов, в демонах ничего нет, их запускаю сам)
interface=eth0
address=
netmask=
broadcast=
gateway=
NETWORK_PERSIST="no"
