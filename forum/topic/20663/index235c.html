Кто-ж знает, что там у вас. Grub2 он как бы самостоятельный, его можно вообще с диска установить на любой диск, но вот в Linux обычно свои скрипты формирующие grub.cfg.
Вот пример что у меня сформировал Mint и Manjaro для Windows 10. Используется поиск по UUID его можно получить с помощью "sudo blkid"
menuentry 'Windows 10 (на /dev/sda1)' --class windows --class os $menuentry_id_option 'osprober-chain-16-значное_число' {
	savedefault
	insmod part_msdos
	insmod ntfs
	set root='hd0,msdos1'
	if [ x$feature_platform_search_hint = xy ]; then
	  search --no-floppy --fs-uuid --set=root --hint-ieee1275='ieee1275//disk@0,msdos1' --hint-bios=hd0,msdos1 --hint-efi=hd0,msdos1 --hint-baremetal=ahci0,msdos1  16-значное_число
	else
	  search --no-floppy --fs-uuid --set=root 16-значное_число
	fi
	parttool ${root} hidden-
	drivemap -s (hd0) ${root}
	chainloader +1
}

menuentry "Windows 10 (на /dev/sda1)" --class windows --class os $menuentry_id_option 'osprober-chain-16-значное_число' {
	insmod part_msdos
	insmod ntfs
	set root='hd0,msdos1'
	if [ x$feature_platform_search_hint = xy ]; then
	  search --no-floppy --fs-uuid --set=root --hint-bios=hd0,msdos1 --hint-efi=hd0,msdos1 --hint-baremetal=ahci0,msdos1  16-значное_число
	else
	  search --no-floppy --fs-uuid --set=root 16-значное_число
	fi
	parttool ${root} hidden-
	drivemap -s (hd0) ${root}
	chainloader +1
}

Кстати, по поводу моей ситуации, удалось установить Windows только после того как удалил все диски.
