Два небольших совета:
1 - используйте паузы между запуском программ.
2 - после КАЖДОЙ строки ставьте знак &.
Пример :
~ $ cat .config/openbox/autostart.sh 
eval `cat $HOME/.fehbg` &
(sleep 5s && volwheel) &
(sleep 8s && clipit) &
(sleep 5s && tint2) &
(sleep 2s && skype) &
(sleep 20s && $HOME/conky_start) &
(sleep 10s && yeahconsole) &
(sleep 25s && xpad) &
(sleep 28s && xxkb) &
(sleep 20s && zim --plugin trayicon) &
## Compositing
(sleep 45s && compton --config ~/.config/compton_nt.conf) &
