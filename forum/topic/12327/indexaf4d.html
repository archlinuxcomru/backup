несколько раз прошла успешно
счас буду удалять FAT32 раздел на SSD

делал так
efibootmgr -d /dev/sdc -p 1 -c -L "Arch Linux-ck" -l /vmlinuz-linux-ck -u "root=/dev/sda2 initrd=/initramfs-linux-ck.img elevator=bfq"
будет
efibootmgr -d /dev/sdc -p 1 -c -L "Arch Linux-ck" -l /vmlinuz-linux-ck -u "root=/dev/sda1 initrd=/initramfs-linux-ck.img elevator=bfq"
