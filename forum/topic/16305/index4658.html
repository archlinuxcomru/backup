reflector.service
[Unit]
Description=Pacman mirrorlist update

[Service]
Type=oneshot
ExecStart=/usr/bin/reflector --verbose -l 5 --sort rate --save /etc/pacman.d/mirrorlist
reflector.timer
[Unit]
Description=Run reflector daily

[Timer]
OnCalendar=12:00
AccuracySec=12h
Persistent=true

[Install]
WantedBy=timers.target
