Ну а мне понадобилось пару часов (да, я долгий, и у меня вечные проблемы с mkinitcpio) для установки базовой системы в VirtualBox без всяких скриптов. Справедливости ради надо отметить, что интернет тут поднимается неимоверно просто, иначе времени понадобилось бы больше. К тому же, систему я практически не настраивал, там даже pacman не работает. Но это уже дело второе.
dhcpcd
ping ya.ru -c 1

cfdisk /dev/sda
mkfs.ext4 /dev/sda1

mount /dev/sda1 /mnt

mkdir -p /mnt/{dev,proc,sys,run,tmp,etc}
mkdir -p /mnt/var/lib/pacman
ls /mnt

vi /etc/pacman.d/mirrorlist
pacman -r /mnt -Sy
pacman -r /mnt -S base
pacman -r /mnt -S grub-bios

mount -o bind /dev /mnt/dev
mount -o bind /proc /mnt/proc
mount -o bind /sys /mnt/sys

chroot /mnt bash

vi /etc/rc.conf
vi /etc/locale.gen
locale-gen

vi /etc/fstab

mkinitcpio -p linux

grub-install /dev/sda
grub-mkconfig -o /boot/grub/grub.cgf

exit
reboot
