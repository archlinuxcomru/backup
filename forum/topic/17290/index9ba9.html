я бы чуть уменьшил гапсы, примерно до 2px
Ну у меня вообще-то spacing и есть по 2 px, наверно можно попробовать до 1 уменьшить или убрать, потому что мне кажется что на mouseResizableTile этот spacing ухудшает захват мышью, делая курсор "хватабельным" в очень узком пространстве
-- layout
myLayoutHook =  avoidStruts
                $ smartBorders
                $ gaps [(U,23)]
                $ onWorkspace (myWorkspaces !! 0) webL
                $ onWorkspace (myWorkspaces !! 1) textL
                $ onWorkspace (myWorkspaces !! 2) termL
                $ onWorkspace (myWorkspaces !! 3) textL
                $ onWorkspace (myWorkspaces !! 4) fullL
                $ onWorkspace (myWorkspaces !! 8) gimpL
                $ standardLayouts

    where
        standardLayouts = (spacing 2 (tiled) ||| spacing 2 (reflectTiled) ||| spacing 2 (Mirror tiled) ||| Circle ||| spacing 2 (Grid) ||| Full ||| tabbed shrinkText myTabConfig)

        tiled   = mouseResizableTile
        reflectTiled = (reflectHoriz mouseResizableTile)
        fullL = (noBorders Full)
        webL = (tabbed shrinkText myTabConfig ||| noBorders Full ||| Circle)
        textL = (tabbed shrinkText myTabConfig ||| spacing 2 (tiled) ||| spacing 2 (reflectTiled))
        termL = (spacing 2 (tiled) ||| spacing 2 (reflectTiled) ||| spacing 2 (Mirror tiled) ||| spacing 2 (Grid))
        gimpL = withIM (0.11) (Role "gimp-toolbox") $ reflectHoriz $ withIM (0.15) (Role "gimp-dock") (noBorders Full ||| tabbed shrinkText myTabConfig)
