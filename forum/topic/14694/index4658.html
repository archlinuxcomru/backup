самба работает. но с др. компов невозможно в расшаренную директорию зайти, хотя сама шара в сетевом окружении видна.
при открытии шары пишет, что отказано в доступе.

Права на папку для шары - 777

Если что, вот мой рабочий конфиг (пускает всех без пароля)

# Global parameters
[global]
        netbios name = FILE_SHARE
        server string = Samba Server on Arch
        security = USER
        map to guest = Bad User
        log file = /var/log/samba/%m.log
        max log size = 50
        load printers = No
        dns proxy = No
        idmap config * : backend = tdb
[SHARE]
        comment = SHARE
        path = /home/srv/SHARE
        read only = No
        guest only = Yes
        guest ok = Yes

права:

$ ls -la  /home/srv/SHARE
