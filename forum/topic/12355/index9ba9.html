Установка TDE/Trinity из бинарных пакетов (не сборка из исходника) / в виртуалке

1. добавляем репу

/etc/pacman.d/mirrorlist # не трогаем, по умолчанию

/etc/pacman.conf
#
# Trintiy/TDE
#
[trinity]
# SigLevel = Never
# SigLevel = Optional TrustAll
# SigLevel = PackageReguire
SigLevel = TrustAll
Server   = http://repo.nasutek.com/arch/contrib/trinity/x86_64/

2. обновляем систему
pacman -Syu

3. установка TDE
pacman -S tde-tdebase
or
pacman -S tde-meta

в той или иной степени проявляется "хрень" в виде "чего-то не хватает, будем ставить так или так, не полностью, или идите вы на ..."

warning: cannot resolve "lcms", a depend of ...
warning: cannot resolve "tde-libkdcraw"
...
lcms, tde-libkdcraw, tde-tdeadmin

при всем при этом, пакеты в репе присутствуют
tde-libkdcraw-14.0.4-1-x86_64.pkg.tar.xz
tde-tdeadmin-14.0.4-1-x86_64.pkg.tar.xz

(хрен с этим lcms, это сторонний пакет, но "родные-то, на месте):

... ладно, худо-бедно соглашаемся на "поставить" то, что есть...

4. Запуск TDE / Косяки

запуск/логин через TDM - ошибка, типа "опа":
could not start tdestartupconfig. Check your installation

TDE - запущена, некоторые уточнения

1. /etc/pacman.conf
[archlinuxcn]
#SigLevel = TrustAll
Server    = https://cdn.repo.archlinuxcn.org/$arch

#
# SigLevel = TrustAll - нужен, иначе "чегой-то" не ставился пакет
# задизеблим, что-бы проверялись подписи
# +
# sudo pacman -S archlinuxcn-keyring
#
and
sudo pacman -S archlinuxcn-keyring

2.install sddm, yay, lcms, libidn, mysql
sudo pacman -S --needed base-devel sddm

sudo pacman -Syu
sudo pacman -S yay

yay -S lcms

sudo pacman -S libidn
cd /usr/lib; sudo ln -sf libidn.so libidn.so.11; cd -

sudo pacman -S mysql # and remove conflict pkg
cd /usr/lib; sudo ln -sf libmysqlclient.so libmysqlclient.so.18; cd -

3.install TDE
sudo pacman -S tde-base

4. run ssdm
sudo systemctl start sdd.service
