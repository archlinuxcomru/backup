Вот моя конфигурация. Шлюз только поправьте под себя, ну и некоторые маршруты уже и не знаю, нужны ли…


$ cat /etc/ppp/ip-up
#!/bin/sh
/sbin/route del $5 dev ppp0
/sbin/route add -host $5 gw 10.210.56.1
/sbin/route add -net 10.0.0.0 netmask 255.0.0.0 gw 10.210.56.1
/sbin/route add -host 83.102.146.96 gw 10.210.56.1
/sbin/route add -host 89.179.135.67 gw 10.210.56.1
/sbin/route del default
/sbin/route add default dev ppp0

$ cat /etc/ppp/ip-down
#!/bin/sh
/sbin/route del $5
/sbin/route del -net 10.0.0.0 netmask 255.0.0.0 gw 10.210.56.1
/sbin/route del -host 83.102.146.96 gw 10.210.56.1
/sbin/route del -host 89.179.135.67 gw 10.210.56.1
/sbin/route del default
/sbin/route add default dev eth0 gw 10.210.56.1

bash-4.0# cat /etc/ppp/peers/corbina
pty “pptp vpn.corbina.net –nolaunchpppd –nobuffer –loglevel 0”
connect /bin/true
name ВАШ_ЛОГИН
remotename PPTP
file /etc/ppp/options_corbina.pptp
ipparam corbina

$ cat /etc/ppp/options_corbina.pptp
lock
noauth
nobsdcomp
nodeflate
nodetach
persist
maxfail 0

bash-4.0# cat /etc/ppp/chap-secrets
ВАШ_ЛОГИН PPTP ВАШ_ПАРОЛЬ *

Права доступа на файлы должны быть следующими -
#chmod 755 /etc/ppp/ip-up
#chmod 755 /etc/ppp/ip-down
#chmod 600 /etc/ppp/peers/corbina
#chmod 644 /etc/ppp/options_corbina.pptp
#chmod 600 /etc/ppp/chap-secrets

запуск из под root
pppd call corbina

Вот такие дела :)
