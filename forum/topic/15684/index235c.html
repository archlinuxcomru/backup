Вот бы как-то обойти это.
Взять скрипт, держать паку src с исходниками и PKGBUILD, и не выделываться yaourt-ом :-)
#!/bin/bash

makepkg -od
source PKGBUILD
[[ $pkgver > $(pacman -Q $pkgname | awk '//{print $2;}') ]] && makepkg -esi
