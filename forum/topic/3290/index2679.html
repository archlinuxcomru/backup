https://wiki.archlinux.org/index.php/Xfce#Running_Xfce
Варианты приводятся самые разные. Я slim не использую, вариант для запуска через xdm
#!/bin/sh
#
# ~/.xinitrc
#
# Executed by startx (run your window manager from here)
if [ -d /etc/X11/xinit/xinitrc.d ]; then
  for f in /etc/X11/xinit/xinitrc.d/*; do
   [ -x "$f" ] && . "$f"
  done
  unset f
fi
exec startxfce4

через startx
#!/bin/sh
#
# ~/.xinitrc
#
# Executed by startx (run your window manager from here)
exec ck-launch-session startxfce4
