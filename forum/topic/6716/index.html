по статье из http://www1.get-e.org/E17_User_Guide/Russian/_pages/print.html сделал так 
mages{
image, "image.jpg"             LOSSY 95;//
}
collections
{
// редактируйте с этого места
group{
name, "desktop/background";
      parts {
         part {
            name,          "background_image";
            mouse_events,  0;
            description {
               state,    "default" 0.0;
               rel1 {
                  relative, 0.0  0.0;
                  offset,   0    0;
               }
               rel2 {
                  relative, 1.0  1.0;
                  offset,   -1   -1;
               }
               image {
                  normal, "image.jpg";
               }
               fill {
                  origin {
                     relative, 0.0 0.0;
                     offset,   0 0;
                  }
                  size {
                     relative, 1.0 1.0;
                     offset,   0 0;
                  }
               }
            }
         }
}
}
создал файл background.edc с этим содержимым, кинул в этот же каталог картинку, в файле подредачил name картинки, запускаю edje_cc -id ./ background.edc background.edj и вот такая ошибка
