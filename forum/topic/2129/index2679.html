Сделал костыль /etc/NetworkManager/dispatcher.d/20-mtu_change
#!/bin/bash
INTERFACE=$1
STATUS=$2
VPN_ID="6cc5eccd-0570-4ed6-89b5-6bf7ad416af5"
if [ "$STATUS" = "vpn-up" -a "$CONNECTION_UUID" = "$VPN_ID" ]; then
    ifconfig "$INTERFACE" mtu 1300
fi
