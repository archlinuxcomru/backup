PKGBUILD`ы прилагаются:
# Maintainer:

pkgname=sparkle
pkgver=1.3
pkgrel=2
pkgdesc="torrent player"
arch=("i386' 'x86_64')
url="https://sparkle.city/"
license=('custom')
depends=('sparkle-vlc', 'sqlite3-unicodesn', 'tufao1')
if [ "${CARCH}" == "x86_64" ];
then
_filearch=amd64
md5sums=('SKIP')
else
_filearch=i386
md5sums=('SKIP')
fi

source=("http://rep.sparkle.city/ubuntu/xenial/pool/${pkgname}_${pkgver}-2_${_filearch}.deb")
md5sums=('SKIP')

package() {
  cd "${srcdir}"
  bsdtar -xf data.tar.xz -C "$pkgdir"
        }
