Для исправления
sudo pacman -Suy
sudo pacman -S freetype2 fontconfig
#для 64 бит мультилиб
 sudo pacman -S lib32-freetype2 lib32-fontconfig
systemctl reboot

Для восстановления иконок
sudo pacman -S librsvg
#для 64 бит мультилиб
sudo pacman -S lib32-librsvg
systemctl reboot
