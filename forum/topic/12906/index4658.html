Вот, до дома добрался. Вас должен интересовать 2й пункт и команда из 1го
У вас задачка несколько другого характера, потому можно написать функцию в .*rc:
pacman() {
if [[ -e /var/lib/pacman/db.lck ]]; then
  DIRECTORY=$(date +%Y%m%d%H%M%S%N)
  mkdir "/tmp/${DIRECTORY}"
  ln -sf "/var/lib/pacman/local" "/tmp/${DIRECTORY}/local"
  ln -sf "/var/lib/pacman/sync" "/tmp/${DIRECTORY}/sync"
  /usr/bin/sudo /usr/bin/pacman --dbpath "/tmp/${DIRECTORY}" $*
  rm -r "/tmp/${DIRECTORY}"
else
  /usr/bin/sudo /usr/bin/pacman $*
fi
}
