не могу загрузиться с iso
Попробуйте добавить в конфиг груба
menuentry 'arch iso' {
  insmod gzio
  insmod part_msdos
  insmod ext2
  set isofile='/boot/iso/archlinux-2020.04.01-x86_64.iso'
  loopback loop $isofile
  linux (loop)/arch/boot/x86_64/vmlinuz archisodevice=/dev/loop0 img_dev=/dev/disk/by-uuid/7b682f29-d895-4195-874d-bf11d4175182 img_loop=$isofile earlymodules=loop
  initrd (loop)/arch/boot/x86_64/archiso.img
}
/boot/iso/archlinux-2020.04.01-x86_64.iso путь до исошки на диске ext4 (без кириллицы в пути)
/dev/disk/by-uuid/7b682f29-d895-4195-874d-bf11d4175182 uuid диска на котором исошка

EDIT 1 - если у тебя grub, то попробуй в grub.cfg прописать для загрузки iso следующее (я гружу так)

submenu "archlinux-2020.02.01-x86_64.iso" {
set root=(hd0,3)
loopback loop (hd0,3)/archlinux-2020.02.01-x86_64.iso
probe -s root_uuid -u $root
linux (loop)/arch/boot/x86_64/vmlinuz img_dev=/dev/disk/by-uuid/$root_uuid img_loop=archlinux-2020.02.01-x86_64.iso earlymodules=loop
initrd (loop)/arch/boot/x86_64/archiso.img
