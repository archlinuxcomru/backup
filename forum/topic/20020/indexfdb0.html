Судя по логу - проблема с файловой системой и при этом похоже еще и полетел superblock.
Можно, конечно, попробовать сразу подсунуть резервный superblock, но лучше идти поэтапно
- скопировать ценную информацию
- если объем не очень большой, можно при желании сделать backup раздела
- выполнить fsck, если superblock испорчен, то проверка завершится с ошибкой
- вот тогда и подсунуть резервный superblock
Ошибки не исчезают с опытом - они просто умнеют

submenu "archlinux-2020.02.01-x86_64.iso" {
 set root=(hd0,3)
 loopback loop (hd0,3)/archlinux-2020.02.01-x86_64.iso
 probe -s root_uuid -u $root
 linux (loop)/arch/boot/x86_64/vmlinuz img_dev=/dev/disk/by-uuid/$root_uuid img_loop=archlinux-2020.02.01-x86_64.iso earlymodules=loop
 initrd (loop)/arch/boot/x86_64/archiso.img

}
