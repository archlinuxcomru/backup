find /usr/share/icons/|grep -i telegram|grep status|sudo xargs rm

sudo gtk-update-icon-cache -f -t '/usr/share/icons/тема-иконок'
rm ~/.cache/icon-cache.kcache
перезайти или перезагрузить DE
