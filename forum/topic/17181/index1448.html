Как вы просматриваете IPTV не прибегая к Firefox? Я бы тоже с удовольствием отбросил бы все бубны...

~ $ yaourt -Qs ace
local/acestream-engine 3.0.5.1-2
    ACE Stream engine
... и установлен aceproxy
запускается это кино вот таким файлом
#!/bin/sh

STATE_DIR=/home/my/aceproxy/log
ACE_LOG=/home/my/aceproxy/log/ace.log
PROXY_LOG=/home/my/aceproxy/log/ace_proxy.log
acestreamengine --client-gtk --bind-all --login LOGIN --password PASS --log-file $ACE_LOG 2>&1 &
echo $! > $STATE_DIR/ace.pid
python2 /home/my/aceproxy/acehttp.py > $PROXY_LOG 2>&1 &
echo $! > $STATE_DIR/proxy.pid

плейлисты обновляются вот так
#!/bin/sh

IPPROXY=192.168.19.22
PORT=8000
path=~/aur
#--- Allfon ---
curl -o $path/allfontv.m3u http://allfon.tv/autogenplaylist/allfontv.m3u
curl -o $path/allfontv.xspf http://allfon.tv/autogenplaylist/allfontv.xspf
cat $path/allfontv.m3u | sed -re "s/acestream:\/\/((.){40})/http:\/\/$IPPROXY:$PORT\/pid\/\1\/s
cat $path/allfontv.xspf | sed -re "s/acestream:\/\/((.){40})/http:\/\/$IPPROXY:$PORT\/pid\/\1\/
sed -i "s/ALLFON.TV//g" $path/allfon.m3u
sed -i "s/ALLFON.TV//g" $path/allfon.xspf
rm $path/allfontv.m3u
rm $path/allfontv.xspf
#--- Torrent-Telik ---
curl -o $path/telik.m3u http://$IPPROXY:$PORT/torrent-telik
#--- TorrentTV SUPER***** ---
curl -o $path/sp.m3u http://super***********/ttv.all.proxy.m3u
curl -o $path/sp.xspf http://super***********/ttv.all.proxy.xspf
sed -i "s/192.168.2.1:82/$IPPROXY:$PORT/g" $path/sp.m3u
sed -i "s/192.168.2.1:82/$IPPROXY:$PORT/g" $path/sp.xspf
потом открываю mc, жмакаю на файл плейлиста - открывается vlc со списком каналов отсортированных по группам
....есть у меня старенький ноут sony (ну очень старенький), там стоит 32-х разрядный arch в минимальной конфигурации без иксов и все это кино, описанное выше, крутится и обновляется по таймеру - я забыл когда в ноут заглядывал последний раз
