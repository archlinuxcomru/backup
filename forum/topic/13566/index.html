Так работает. asound.conf:

defaults.pcm.!rate_converter "samplerate_best"
defaults.pcm.dmix.!rate 44100
defaults.pcm.dmix.!format "unchanged"

pcm.hdmi_formatted {
  type plug
  slave.pcm "dmix:0,9"
  hint {
    show on
    description "HDMI Out"
  }
}

pcm.hdmi_output {
  type softvol
  slave.pcm hdmi_formatted
  control.name hdmi_volume
  control.card 0
}

pcm.!default hdmi_output

А так собственно тема. Хромиум рыгает что-то там про неправильный формат:

defaults.pcm.!rate_converter "samplerate_best"
defaults.pcm.dmix.!rate 96000
defaults.pcm.dmix.!format S24LE

pcm.hdmi_formatted {
  type plug
  slave.pcm "dmix:0,9"
  hint {
    show on
    description "HDMI Out"
  }
}

pcm.hdmi_output {
  type softvol
  slave.pcm hdmi_formatted
  control.name hdmi_volume
  control.card 0
}

pcm.!default hdmi_output
