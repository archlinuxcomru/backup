Я его и выложил специально для того, чтобы кто- то из более опытных пользователей проверил, все ли там правильно.
Подправил все метки, дабы исключить такое в будущем.
# по-умолчанию выбран пункт меню 0
set default=0

# разрешение экрана
set gfxmode=console

# управляет видеорежимом, в котором загружается ядро Linux
set gfxplayload=text

# загрузить модули поддержки графического режима для uefi
insmod efi_gop
insmod efi_uga

# при бездействии пользователя загрузка через n секунд
set timeout=3

# пункт меню номер 0
menuentry "ArchLinux" {
 insmod part_gpt
 insmod ext2
 set root='hd1,gpt2'
 linux /boot/vmlinuz-linux-lts root=UUID=df8c5030-729c-47ee-b2e3-5db8c49be422 rw audit=0 mitigations=off #nvidia-drm.modeset=1
 initrd /boot/initramfs-linux-lts.img
}

# пункт меню номер 1
menuentry "ArchLinuxFallback" {
 insmod part_gpt
 insmod ext2
 set root='hd1,gpt2'
 linux /boot/vmlinuz-linux-lts root=UUID=df8c5030-729c-47ee-b2e3-5db8c49be422 rw audit=0
 initrd /boot/initramfs-linux-lts-fallback.img
}

# пункт меню номер 2
menuentry "ArchLinux_btrfs" {
 insmod part_gpt
 insmod btrfs
 set root='hd0,gpt2'
 linux /root/boot/vmlinuz-linux-lts root=UUID=82a15126-db71-4d16-8137-3d760d33d8ae rw rootflags=subvol=root audit=0 mitigations=off #nvidia-drm.modeset=1
 initrd /root/boot/initramfs-linux-lts.img
}

# пункт меню номер 3
menuentry "ArchLinux_btrfsFallback" {
 insmod part_gpt
 insmod btrfs
 set root='hd0,gpt2'
 linux /boot/vmlinuz-linux-lts root=UUID=82a15126-db71-4d16-8137-3d760d33d8ae rw rootflags=subvol=root audit=0
 initrd /boot/initramfs-linux-lts-fallback.img
}
