conky.config = {

--own_window_type = 'normal',

no_buffers = true,

-- Graphics
use_spacer = 'none',
use_xft = true,
font = 'DejaVu Sans Mono:size=12',
update_interval = 1.0,
uppercase = false,
stippled_borders = 0,
border_width = 1,
draw_borders = false,
draw_graph_borders = false,
draw_outline = false,
draw_shades = false,
show_graph_scale = false,
show_graph_range = false,

-- Layout

gap_x = 25,

minimum_height = 250,
minimum_width = 250,
maximum_width = 270,
net_avg_samples = 1,
cpu_avg_samples = 6,
out_to_console = false,

extra_newline = false,

-- Graphics Colors

}
