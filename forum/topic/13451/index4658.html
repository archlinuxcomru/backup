РЕШЕНИЕ


cryptsetup luksFormat /dev/sdaX
пишем YES большими буквами и вводим 2 раза пароль

cryptsetup open --type luks /dev/sdaX lvm
вводим пароль

pvcreate /dev/mapper/lvm
vgcreate <имя гр. томов> /dev/mapper/lvm
lvcreate -L <размер+G,M,K> <имя гр. томов> -n <имя тома>
lvcreate -L <размер+G,M,K> <имя гр. томов> -n <имя тома>
lvcreate -L <размер+G,M,K> <имя гр. томов> -n <имя тома>
mkfs.ext4 /dev/mapper/<имя гр. томов>тире<имя тома>
mkfs.ext4 /dev/mapper/<имя гр. томов>тире<имя тома>
mkswap /dev/mapper/<имя гр. томов>тире<имя тома>
mount /dev/<имя гр. томов>/<имя тома> /mnt
mkdir /mnt/home
mkdir /mnt/boot
mount /dev/<имя гр. томов>/<имя тома> /mnt/home
mount /dev/sdaХ /mnt/boot загр. необходимо расположить на незашифрованном разделе


swapon /dev/<имя гр. томов>/<имя тома>
pacstrap /mnt base base-devel
genfstab -p /mnt > /mnt/etc/fstab
arch-chroot /mnt
nano /etc/pacman.conf
необходимо стереть

[multilib]
Include = /etc/pacman.d/mirrorlist


nano /etc/hostname
указываю имя комп.

ln -s /usr/share/zoneinfo/Europe/Moscow /etc/localtime
nano /etc/locale.gen
стираю комменты перед En и RU

locale-gen
nano /etc/locale.conf(LANG="ru_RU.UTF-8 UTF-8")
passwd пароль root-a
создаю уч. запись

useradd -m -g users -s /bin/bash zer
passwd zer пароль zer-a
gpasswd -a zer audio
gpasswd -a zer games
gpasswd -a zer lp
gpasswd -a zer optical
gpasswd -a zer power
gpasswd -a zer scanner
gpasswd -a zer storage
gpasswd -a zer video
EDITOR=nano visudo
дописываю
zer ALL=(ALL) ALL)


nano /etc/mkinitcpio.conf
в HOOKS между blok и filesystem пишу encrypt lvm2


pacman -S grub-bios
Если на комп. стоят и другие ОС уст.
os-prober


grub-install /dev/sda
root=/dev/mapper/arch-root
nano /etc/default/grub
в GRUB_CMDLINE_LINUX_DEFAULT= пишем"cryptdevice=UUID=<uuid зашифр. раздела>:<имя гр. томов>"

mkinitcpio -p linux
grub-mkconfig -o /boot/grub/grub.cfg
exit
