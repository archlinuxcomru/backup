конфиг самбы
#======================= Global Settings =====================================
[global]
   workgroup = WORKGROUP
   server string = Arch Linux Samba Server
   netbios name = DellVostro1700
   security = user
   usershare path = /var/lib/samba/usershares
   usershare allow guests = yes
   usershare max shares = 100
   usershare owner only = no
   map to guest = Bad User

   printcap name = /etc/printcap
   load printers = yes

   log file = /var/log/samba/%m.log
   max log size = 50

   dns proxy = no

#============================ Share Definitions ==============================
[printers]
   comment = All Printers
   path = /var/spool/samba
   browseable = no
# Set public = yes to allow user 'guest account' to print
   guest ok = yes
   writable = no
   printable = yes

stmc, я не совсем понял с чем у вас проблемы. Вы не можете зайти на свой комп с другого с помощью нау? Если так, то проблему нужно искать в первую очередь в настройках samba (при условии, что нет всяких файрволов и прочей мути).
Если вы не можете просматривать со своего компьютера - samba не причем. Для такого случая достаточно одного smbclient (ну и, возможно - gvfs-smb). Именно на эти два пакета и обратите внимание.
