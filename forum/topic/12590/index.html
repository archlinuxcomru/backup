собственно, имеются 2 виртуалки, запускаемые так:
xp               = "VBoxSDL --fullscreenresize --startvm xp --nofstoggle --termacpi"
arch             = "VBoxSDL --fullscreenresize --startvm arch --nofstoggle --termacpi"
xpop= $( xprop | grep --color=none "WM_CLASS\|^WM_NAME" | xmessage -file - )
определяет их так:
WM_NAME(STRING) = "xp - Oracle VM VirtualBox"
WM_CLASS(STRING) = "VBoxSDL", "VBoxSDL"
и
WM_NAME(STRING) = "arch - Oracle VM VirtualBox"
WM_CLASS(STRING) = "VBoxSDL", "VBoxSDL"

необходимо, чтобы виртуалки находились на 2 разных тэгах.
в rc.lua прописал:
{ rule = { class    = "VBoxSDL" , name = "xp - Oracle VM VirtualBox"                   }, properties = { tag = tags[1][6], switchtotag = false } },
{ rule = { class    = "VBoxSDL" , name = "arch - Oracle VM VirtualBox"                }, properties = { tag = tags[1][7], switchtotag = false } },
но они стартуют на активном тэге. ЧЯДНТ?
