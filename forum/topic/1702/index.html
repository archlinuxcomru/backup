Все настройки для русификации в /etc/rc.conf
Вот огрызок от моего конфига. Думаю, у тебя должно быть что либо подобное.
#
LOCALE="ru_RU.UTF-8"
DAEMON_LOCALE="no"
HARDWARECLOCK="localtime"
TIMEZONE="Europe/Moscow"
KEYMAP="ruwin_cplk-UTF-8"
CONSOLEFONT="UniCyr_8x16"
CONSOLEMAP=
USECOLOR="yes"
