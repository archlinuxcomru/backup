depends=('scons' 'smpeg' 'sdl' 'sdl_image' 'zlib' 'expat' 'lua' 'libvorbis')

  cd $startdir/src/$pkgname-$pkgver
  scons prefix=$startdir/pkg/usr || return 1
  scons install
}
