#include <vector>
#include <ctime>
#include <random>
#include <algorithm>
#include <iostream>

std::vector<int> getRandMatrix(int size) {
    std::mt19937 gen;
    gen.seed(static_cast<unsigned int>(time(0)));
    std::vector<int> vec(size);
    for (int  i = 0; i < size; i++) { vec[i] = gen() % 100; }
    return vec;
}
