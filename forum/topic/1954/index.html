Это не аплеты… а упорядочить можно системные индикаторы: a11y, клавиатура, звук, блютус,батарея, сеть, меню пользователя.
sudo gedit /usr/share/gnome-shell/js/ui/panel.js
строка 35 : 
const STANDARD_STATUS_AREA_ORDER = ['keyboard', 'volume', 'battery','bluetooth', 'network', 'userMenu'];
меняете порядок…
Удачи!

з.ы. у меня в примере a11y удалено совсем… если надумаете так сделать, то не забудьте закоментить ниже:
const STANDARD_STATUS_AREA_SHELL_IMPLEMENTATION = {
    //'a11y': imports.ui.status.accessibility.ATIndicator,
    'volume': imports.ui.status.volume.Indicator,
    'battery': imports.ui.status.power.Indicator,
    'keyboard': imports.ui.status.keyboard.XKBIndicator,
    'userMenu': imports.ui.userMenu.UserMenuButton
