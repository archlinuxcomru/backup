## -- snd_virtuoso --
alias snd-card-0 snd_virtuoso
alias snd-slot-0 snd_virtuoso
options snd_virtuoso index=0 id=DS
## -- snd_hda_intel --
alias snd-card-1 snd_hda_intel
alias snd-slot-1 snd_hda_intel
options snd_hda_intel index=1 id=NVidia
## -- snd_usb_audio --
alias snd-card-2 snd_usb_audio
alias snd-slot-2 snd_usb_audio
options snd_usb_audio index=2 id=U0x46d0x807
