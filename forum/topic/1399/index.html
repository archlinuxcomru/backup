всем спасибо за внимание, проблема решена.
mkdir /a
mount /dev/sdaX /a
cp -R /a/lib/* /a/usr/lib
rm -fr /a/lib
pacman -S glibc -r /a --cachedir /var/chache/pacman/pkg
