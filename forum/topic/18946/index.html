или тоже самое но в конфиге иксов

Section	"InputClass"
    Identifier         "system-keyboard"
    MatchIsKeyboard    "on"
    Option             "XkbLayout" "us,ru"
    Option             "XkbModel" "pc105"
    Option             "XkbVariant" "-"
    Option             "XkbOptions" "terminate:ctrl_alt_bksp,grp:alt_shift_toggle,grp_led:scroll"
EndSection
