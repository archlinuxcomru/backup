А как информацию об обновлениях в коньки выводить? И работает ли это с пактеами из AUR?

конфиг для conky и скрипт был предоставлен snoopcatt

Статус:${alignr}${execi 1800 [ -f /var/lib/pacman/db.lck ] && echo Идет операция || cat /var/conky/alpm|grep stat|cut -d '=' -f 2-99}
Пакетов:${alignr}${execi 1800 [ -f /var/lib/pacman/db.lck ] && echo Идет операция || cat /var/conky/alpm|grep count|cut -d '=' -f 2-99}
Размер:${alignr}${execi 1800 [ -f /var/lib/pacman/db.lck ] && echo Идет операция || cat /var/conky/alpm|grep sz|cut -d '=' -f 2-99}
В /var/conky/alpm пишет скрипт по крону каждый час

#!/bin/bash
[ "`whoami`" != "root" ] && exit 1
txt='/var/conky/alpm'
echo "date=`date +%d.%m\ в\ %H:%M`" > $txt
isInetOk()
{
  ping -c 1 google.com >/dev/null 2>/dev/null && echo 0 || echo 1
}
if [ "`isInetOk`" = '0' ] 
 then
pacman -Sy >/dev/null 2>&1
count=`pacman -Sup | grep 'pkg'| wc -l`
echo "count=$count" >> $txt
 if [ "$count" != "0" ] 
  then
   echo "stat=Есть обновления" >> $txt 
   echo "sz=`yes n 2>/dev/null | pacman -Su 2>/dev/null|grep загружаемых|cut -d ':' -f 2|awk '{ print $1 " " $2 }'`" >> $txt
  else
   echo "stat=Нет обновлений" >> $txt
   echo "sz=0 Мб" >> $txt
 fi
fi
if [ "`isInetOk`" = '1' ] 
 then
echo "stat=Ошибка сети" >> $txt
echo "count=Неизвестно" >> $txt
echo "sz=Неизвестно" >> $txt
fi
