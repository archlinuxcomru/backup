print_trans(){
rt_c="timeout 1 xmlrpc localhost:9091 d.multicall "main""
if [[ -n `$rt_c "d.get_state=" | grep 'integer: 1'` ]]; then
  _up_tr=(`$rt_c "d.get_up_rate=" | grep integer: | awk '{print $5}'`)
  _down_tr=(`$rt_c "d.get_down_rate=" | grep integer: | awk '{print $5}'`)
  for i in $(seq 0 `echo ${_down_tr[*]} | wc -w`); do
    if [[ -n ${_up_tr[$i]} ]]; then
      let up_tr=$up_tr+${_up_tr[$i]}
      let down_tr=$down_tr+${_down_tr[$i]}
    fi
  done
  if (( $down_tr > $up_tr )); then
    echo -ne "\x03⇓\x04`echo $down_tr/1000 | bc`\x01|"
  else
    echo -ne "\x03⇑\x04`echo $up_tr/1000 | bc`\x01|"
  fi
fi
}

#!/bin/bash

IF=enp2s0

R1=`cat /sys/class/net/$IF/statistics/rx_bytes`
T1=`cat /sys/class/net/$IF/statistics/tx_bytes`
sleep 1
R2=`cat /sys/class/net/$IF/statistics/rx_bytes`
T2=`cat /sys/class/net/$IF/statistics/tx_bytes`
TBPS=`expr $T2 - $T1`
RBPS=`expr $R2 - $R1`
TKBPS=`expr $TBPS / 1024`
RKBPS=`expr $RBPS / 1024`
echo "⇣$RKBPS kb/s ⇡$TKBPS kb/s"
