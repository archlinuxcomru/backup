print_date(){
if [[ `cat $path_dir/var_time` == 1 ]]; then
       _time=" "  && echo 0 > $path_dir/var_time
else
       _time=":"  && echo 1 > $path_dir/var_time
fi
echo -ne "`date +%H"$_time"%M`"
}

print_load(){
_load=`cat /proc/loadavg | awk '{print $1}' | grep -o '^[^.]*'`
if (( $_load > 2 )); then
        _load="красный цвет"
else
        _load="обычный цвет"
fi
echo -ne "$_load|"
}

while true; do
    xsetroot -name "$(print_date)$(print_load)"
    sleep 1
done
