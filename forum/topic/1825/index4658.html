etc/ppp/options-mobile
/dev/ttyUSB0
921600
defaultroute
usepeerdns
crtscts
lock
noauth
local
persist
modem
nopcomp
novjccomp
nobsdcomp
nodeflate
noaccomp
ipcp-accept-local
ipcp-accept-remote
noipdefault

/etc/ppp/peers/Megafon
file /etc/ppp/options-mobile
connect "/usr/sbin/chat -v -t15 -f /etc/ppp/chatscripts/Megafon.chat"

ABORT 'BUSY'
ABORT 'NO CARRIER'
ABORT 'VOICE'

ABORT 'NO DIAL TONE'
ABORT 'NO ANSWER'

'OK-AT-OK' 'ATI'
'OK' 'ATZ'
