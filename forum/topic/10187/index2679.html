Собсно rsync с rsync.archlinux.org последнее время недоступен.
рсинкатся можно с mir1.archlinuxfr.org и mir2.archlinux.org

как пример - скрипт рсинка который пользую я. (мне нужны только i686)

rsync -avz --delete --exclude=os/x86_64 mir1.archlinuxfr.org::archlinux/current /mirror/current
rsync -avz --delete --exclude=os/x86_64 mir1.archlinuxfr.org::archlinux/extra/os /mirror/extra
rsync -avz --delete --exclude=os/x86_64 mir1.archlinuxfr.org::archlinux/community /mirror/community
rsync -avz --delete --exclude=os/x86_64 mir1.archlinuxfr.org::archlinux/testing /mirror/testing
rsync -avz --delete --exclude=os/x86_64 mir1.archlinuxfr.org::archlinux/unstable /mirror/unstable
logger -t Mirroring -s "Sync Complite"
