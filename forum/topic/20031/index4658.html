После того, как ноут выходит из спящего режима
Навсякий случай, может пригодится --- если наблюдаются проблемы после выхода из suspend/hibernate, определяют чем обусловлена проблема (в большинстве случаев это связано с модулями) и пишется скрипт /usr/lib/systemd/system-sleep/name.sh, в котором указываются нужные действия перед входом в suspend (например, выгрузка модуля) и после выхода из suspend (например, загрузка модуля) … типа такого
#!/bin/sh
case $1/$2 in
  pre/*)
    /usr/bin/modprobe -r module  ### любая команда или скрипт
    ;;
  post/*)
    /usr/bin/modprobe module  ### любая команда или скрипт
    ;;
esac
