build() {
  tar xzf $startdir/src/kaffeine-sc_0.3.9svn.orig.tar.gz
  cd $startdir/src/$pkgname-$pkgver
  ./configure --prefix=/opt/kde
  make || return 1
  make prefix=$startdir/pkg/opt/kde install
}
