Сегодня с утра увидел на ЛОРе новость про выход свежего Лиська…
Вообще Лиськ терпеть не мог раньше за убогий интерфейс… а это думаю “хм.. на QT4 написали.. может чего хорошего вышло”. И задумал его поставить.

К сожалению в репозитариях еще 1.3.6 нет, и даже в АУРе тоже нету….
В общем я рискнул попробовать собрать из исходников!

И каково было мое удивление, когда немного правленный (наугад!) PKGBUILD от 1.3.5 скомпилял рабочую версию!
Ура Арчу! Я с каждым днем все больше и больше люблю этот дистр :) ))


ЗЫ Для желающих опробовать свежий ЛИСЬК предлагаю правленный PKGUILD:
—PKGBUILD—
# $Id: PKGBUILD 5779 2008-07-20 21:00:46Z jgc $
# Maintainer: Juergen Hoetzel <[email protected]>;
# thnx to Juergen for original PKGBUILD.
# adopted for 1.3.6. by point212
pkgname=licq
pkgver=1.3.6
pkgrel=1
pkgdesc=“Advanced graphical ICQ clone and more for Unix”
arch=(i686 x86_64)
url="http://www.licq.org"
license=('GPL')
depends=('libxss' ‘qt4’ ‘openssl’)
options=(!libtool)
source=( http://downloads.sourceforge.net/source … icq/licq-$ {pkgver}.tar.bz2)
build() {
cd ${srcdir}/${pkgname}-${pkgver}
. /etc/profile.d/qt3.sh

# licq
GPGME_CONFIG=/bin/false ./configure –prefix=/usr || return 1
make || return 1
make DESTDIR=${pkgdir} install || return 1

# qt-gui
cd plugins/qt-gui || return 1
# for some reason, –without-kde makes ./configure think that
# the QT libs are missing. ???
GPGME_CONFIG=/bin/false ./configure –prefix=/usr || return 1
make || return 1
cd po || return 1
lrelease *.ts || return 1
cd ..
make DESTDIR=${pkgdir} install || return 1
rm -rf ${pkgdir}/opt
}
