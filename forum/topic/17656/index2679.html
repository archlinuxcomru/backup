спасибо вам XX1232, давно хотел разобраться в этом вопросе всё руки не доходили, всё ниже написанное есть в wiki правда в разных статьях.
в pulse, раскоментируем и убираем авторизацию для под сети (на сервере и клиенте)
$ cat /etc/pulse/default.pa
load-module module-native-protocol-tcp auth-ip-acl=192.168.0.0/24 auth-anonymous=1
mpd - локальный
$ cat ~/.mpd/mpd_local.conf
bind_to_address "~/.mpd/mpd.socket"
db_file            "~/.mpd/database"
log_file           "~/.mpd/log"
audio_output {
  type            "pulse"
  name            "mpd_pulse"
}
удалённый
$ cat ~/.mpd/mpd_remote.conf
bind_to_address "0.0.0.0"
port "6600"
db_file            "~/.mpd/database"
log_file           "~/.mpd/log_user"
audio_output {
  type            "pulse"
  name            "mpd_pulse_user"
  server          "адрес pulse сервера"
}
systemd
$ cat ~/.config/systemd/user/[email protected]
[Unit]
Description=Music Player Daemon %I
[Service]
ExecStart=/usr/bin/mpd --no-daemon /home/user/.mpd/mpd_%I.conf
[Install]
WantedBy=default.target
