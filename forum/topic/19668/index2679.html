зависит от производительности проца
Недавно на лоре обсуждали тест скорости чтения рам в зависимости от блока, в принципе он и подошел бы для оценки теоритической скорости
. комментарии файла ioblksize.h (coreutils) от автора его кода Jim Meyering
#!/bin/bash
for i in $(seq 0 10)
	do bs=$((1024*2**$i))
    printf "%7s=" $bs
    timeout --foreground -sINT 2 \
        dd bs=$bs if=/dev/zero of=/dev/null 2>&1 \
        | sed -n 's/.* \([0-9,.]* [GM]B\/s\)/\1/p'
done
