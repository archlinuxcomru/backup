настроить .xinitrc не вышло.
Скиньте кто-нибудь рабочий пример его
#!/bin/sh

userresources=$HOME/.Xresources
usermodmap=$HOME/.Xmodmap
sysresources=/etc/X11/xinit/.Xresources
sysmodmap=/etc/X11/xinit/.Xmodmap

# merge in defaults and keymaps

if [ -f $sysresources ]; then
    xrdb -merge $sysresources

fi

if [ -f $sysmodmap ]; then
    xmodmap $sysmodmap
fi

if [ -f "$userresources" ]; then
    xrdb -merge "$userresources"

fi

if [ -f "$usermodmap" ]; then
    xmodmap "$usermodmap"
fi
#
# start some nice programs

if [ -d /etc/X11/xinit/xinitrc.d ] ; then
 for f in /etc/X11/xinit/xinitrc.d/?*.sh ; do
  [ -x "$f" ] && . "$f"
 done
 unset f
fi

#exec startplasma-x11
#exec startlxde
#exec io.elementary.cerbere
# startdde
#export XDG_SESSION_TYPE=x11
#export GDK_BACKEND=x11
#exec startxfce4
#export XDG_CURRENT_DESKTOP=Pantheon
#export GNOME_SHELL_SESSION_MODE=pantheon
#exec gnome-session --session=pantheon
export XDG_SESSION_TYPE=x11
exec cinnamon-session

If the Cinnamon (Software Rendering) session is required, use cinnamon-session-cinnamon2d instead of cinnamon-session.
