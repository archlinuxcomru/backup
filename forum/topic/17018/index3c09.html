Но хотелось бы, просто пкм на панели.
Это в другое место совсем вписывать надо. И там ПКМ обычно занята. Нужное место:

-- Configure buttons for tasklist
local tasklist_buttons = gears.table.join(
    awful.button({ }, 1, function (c)
        if c == client.focus then
            c.minimized = true
        else
            -- Without this, the following
            -- :isvisible() makes no sense
            c.minimized = false
            if not c:isvisible() and c.first_tag then
                c.first_tag:view_only()
            end
            -- This will also un-minimize
            -- the client, if needed
            client.focus = c
            c:raise()
        end
    end),

    awful.button({ }, 3,  client_menu_toggle() ),
    awful.button({ }, 4, function () awful.client.focus.byidx(-1) end),
    awful.button({ }, 5, function () awful.client.focus.byidx( 1) end)
)

P.S.: Можно на вторую кнопку повесить...
