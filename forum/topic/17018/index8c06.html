Диаграмки дисков (хотя чего угодно можно прикрутить...)


arcchart = wibox.widget.piechart()
arcchart.colors = {
  color1,
  color2,
  color3
  }


fs = lain.widget.fs({
    options = "--exclude-type=tmpfs",
    settings  = function()
        arcchart.data_list = {
          {"Free", ((tonumber(fs_info["/ avail_p"]) + tonumber(fs_info["/home avail_p"])) - (tonumber(fs_info["/ used_p"])+tonumber(fs_info["/home used_p"])))},
          {"/", tonumber(fs_info["/ used_p"])},
          {"/home", tonumber(fs_info["/home used_p"])}
        }
    end
})




arcchart = wibox.container.arcchart()
arcchart.colors = {
  color1,
  color2,
  color3
  }


fs = lain.widget.fs({
    options = "--exclude-type=tmpfs",
    settings  = function()
    arcchart.values = {
          ((tonumber(fs_info["/ avail_p"]) + tonumber(fs_info["/home avail_p"])) - (tonumber(fs_info["/ used_p"])+tonumber(fs_info["/home used_p"]))),
          tonumber(fs_info["/ used_p"]),
          tonumber(fs_info["/home used_p"])
        }
