Можно. Прям разные для всех и прям в rc.lua. Например:

    layouts = awful.layout.layouts
    tags = {
        names  = { "⠐", " ⠡ ", " ⠲ ", " ⠵ ", " ⠾ ", " ⠿ " },
        layout = { layouts[2], layouts[4], layouts[3], layouts[5], layouts[1], layouts[6] }

    for s = 1, screen.count() do
        tags[s] = awful.tag(tags.names, s, tags.layout)
    end
