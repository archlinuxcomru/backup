Если действовать в соответствии с этим и не устанавливать один layout для всех тегов, как показано в первом же примере там, а добавлять теги по отдельности, то в итоге мы и получим "аномальную зону" (проверено на дефолтном конфиге). Не, ну их, аномалии эти.
Пока остановился на такой конструкции:
local names = { "Ƅ", "ƀ", "Ɵ", "ƈ", "Ɗ" }
local l = awful.layout.suit  -- Just to save some typing: use an alias.
local layouts = { l.tile.bottom, l.tile.bottom, l.tile.bottom, l.spiral.dwindle, l.floating }
for s = 1, screen.count() do
awful.tag(names, s, layouts)
awful.tag.setncol( 4, nil, true)
end
