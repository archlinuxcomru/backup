Только сразу не повторяйте, подождите до полудня, чтобы если я где ошибся, меня поправили/добавили

mkdir -p /mnt/badarch
mount /dev/sdaX /mnt/badarch
cp -R /mnt/badarch/lib/* /mnt/badarch/usr/lib
cp -R /mnt/badarch/lib32/* /mnt/badarch/usr/lib32
rm -R /mnt/badarch/lib*
pacman -Sy
pacman -S glibc -r /mnt/badarch
pacman -S linux bash -r /mnt/badarch
mount --bind /proc /mnt/badarch/proc
mount --bind /dev /mnt/badarch/dev
mount --bind /sys /mnt/badarch/sys
chroot /mnt/badarch
pacman -Syu
mkinitcpio -p linux
exit
umount /mnt/badarch/proc
umount /mnt/badarch/dev
umount /mnt/badarch/sys
umount /mnt/badarch
единственно что не описано, подключение к интернету и монтирования др. каталогов, которые висят на отдельных разделах, их по-хорошему тоже надо примонтировать в иерархию /mnt/badarch .
