сегодня решил изменить конфиг, вот последним сообщением, для примера. Ну вроде рабочий.
Ессно последний пакмен и pacman-key … перед обновлением.
[options]
HoldPkg     = pacman glibc
SyncFirst   = pacman
Architecture = auto
CheckSpace
SigLevel = Never
[core]
SigLevel = PackageRequired
Include = /etc/pacman.d/mirrorlist
[extra]
SigLevel = PackageRequired
Include = /etc/pacman.d/mirrorlist
[community]
SigLevel = PackageRequired
Include = /etc/pacman.d/mirrorlist
[aurbuild]
Server = http://abs.igus.lv/aurbuild/$arch
#[archlinuxfr]
#Server = http://repo.archlinux.fr/i686
