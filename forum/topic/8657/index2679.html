Для начала привяжи интерфейсы к макам, а то плавfть при перезагрузке могут спокойно
/etc/rules.d/75-user-net.rules
# eth0
SUBSYSTEM=="net", DRIVERS=="?*", ATTRS{address}=="00:15:17:27:6f:49", NAME="eth0"
#eth1
SUBSYSTEM=="net", DRIVERS=="?*", ATTRS{address}=="00:15:17:27:6f:47", NAME="eth1"

INTERFACES=(eth0 !eth1)
