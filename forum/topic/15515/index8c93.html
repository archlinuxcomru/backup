export DISPLAY=:0
export MALLOC_CHECK_=0
KWIN_EXPLICIT_SYNC=0 kwin_x11 --replace &

Первый экспорт - поскольку запускал изначально при зависании плазмы из другого экрана, второй - что бы плазма не валилась при ошибках Xorg.
