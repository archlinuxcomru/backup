-- moc ===============================================================

	local moc_bar = wibox.widget {
        max_value     = 1,
        forced_height = 5,
        forced_width  = beautiful.left_panel_width,	-- 100,
        paddings      = 1,
        border_width  = 1,
        border_color  = beautiful.border_color,
        ticks = true,
        widget        = wibox.widget.progressbar,
    }
local text_moc = wibox.widget.textbox()

local mymoc = lain.widgets.contrib.moc({
	followmouse = true,
	settings = function()
		moc_bar:set_value(moc_now.cs / moc_now.ts)
		text_moc:set_markup(markup('#aebdae',"MOC" .. '\n' .. "State : ".. moc_now.state .. '\n' ..
											"Artist : " .. moc_now.artist .. "\n" ..
											"Title : " .. moc_now.title

							))
		text_moc:set_align("center")

    end
	})

my_moc = wibox.container.margin(mymoc, 3, 3, 3, 3)

textmocbg = wibox.container.background(text_moc, "#00000045", gears.shape.rectangle)
mytextmoc = wibox.container.margin(textmocbg, 3, 3, 3, 0)
