Можно через gconf-editor, секция system - storage - default_options
можно скриптик на будущее заделать
#!/bin/bash
# Опции монтирования гнома
VFAT="uid=,quiet,iocharset=utf8,codepage=1251,umask=0,flush"
ISO9660="uid=,iocharset=utf8"
NTFS3G="uid=,quiet,locale=ru_RU.utf8,dmask=000,fmask=111"
TOOLS="gconftool-2 --set"
echo "* Параметры подключения по умолчанию для файловой системы *"
$TOOLS /system/storage/default_options/vfat/mount_options --type list --list-type=string "[${VFAT}]"
$TOOLS /system/storage/default_options/iso9660/mount_options --type list --list-type=string "[${ISO9660}]"
$TOOLS /system/storage/default_options/ntfs-3g/mount_options --type list --list-type=string "[${NTFS3G}]"
exit 0
