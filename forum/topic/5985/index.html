#!/bin/bash
case $1 in
start)
    echo "off"
        echo 0 > /sys/devices/platform/asus_laptop/wlan
        ;;
stop)
    echo "on"
        echo 1 > /sys/devices/platform/asus_laptop/wlan
        ;;
restart)
    $0 stop
    $0 start
    ;;
*)
    echo "Usage: $0 {start|stop|restart}" >&2
    exit 1
esac

я этот демон при старте системы пускаю и asus_laptop загружен и светодиойд сам гаснет.
Про Fn 2 не знаю но можно эту строку в visudo(чтобы пароль не спрашивал) прописать а потом подвесить выполнение строки на какую не будь кнопку (это через wm) кстати этот 0 Tx-Power не отключает.
