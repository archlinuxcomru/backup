Если на будущее понадобиться подправить AppImage

#создаем папку и переходим в нее
mkdir /tmp/newAppImage
cd /tmp/newAppImage
#распаковываем AppImage
'/home/oleg/Disk-nx/oleg/Загрузки/Deepnest-1.0.5-linux/Deepnest-1.0.5-x86_64.AppImage' --appimage-extract
#добавляете или правите нужное в /tmp/newAppImage/squashfs-root
#либы например положить в /tmp/newAppImage/squashfs-root/usr/lib
#и собираете обратно
#качаем сборщик
wget "https://github.com/AppImage/AppImageKit/releases/download/continuous/appimagetool-x86_64.AppImage"
chmod a+x appimagetool-x86_64.AppImage
#собираем
./appimagetool-x86_64.AppImage /tmp/newAppImage/squashfs-root
#копируем собранный AppImage куда надо
#например
mv '/tmp/newAppImage/Deepnest-x86_64.AppImage' '/home/oleg/.local/bin/Deepnest-x86_64.AppImage'
# запускаем
'/home/oleg/.local/bin/Deepnest-x86_64.AppImage'

Почитать
https://github.com/AppImage/AppImageKit/blob/master/README.md
