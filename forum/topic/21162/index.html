скрипт
Макет (wav):
#!/usr/bin/bash
cd input_dir
for i in *
do out="${i%.*}"
ffmpeg -i "$i" -c:a pcm_s16le "/output_dir/$out.wav"
done
ну или в mp3:
---
ffmpeg -i "$i" -c:a libmp3lame -b:a 64k "/output_dir/$out.mp3"
---
