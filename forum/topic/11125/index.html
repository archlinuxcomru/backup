# echo 5 > /sys/class/backlight/acpi_video0/brightness
$ xbacklight -set 5
# setpci -s 00:02.0 f4.b=a0

Добавлял к загрузке ядра параметры:
acpi_backlight=vendor
acpi_osi=Linux acpi_backlight=vendor
acpi_osi=Linux acpi_backlight=legacy
acpi_osi=Linux
С последними двумя заработали клавиши Fn + стрелки.

Kycok, так тоже LMT подсветку не регулирует? /etc/laptop-mode/conf.d/lcd-brightness.conf:
CONTROL_BRIGHTNESS=1
BATT_BRIGHTNESS_COMMAND="echo 3"
LM_AC_BRIGHTNESS_COMMAND="echo 8"
NOLM_AC_BRIGHTNESS_COMMAND="echo 8"
BRIGHTNESS_OUTPUT="/sys/class/backlight/acpi_video0/brightness"
"echo [значение]" подобрать под себя, разумеется. И после правки конфига вроде как нужно перезапустить сервис:
# systemctl restart laptop-mode.service
