# Contributor: arkhnchul <[email protected]>;
pkgname=qutim
pkgver=0.3
pkgrel=1
pkgdesc="Multiprotocol IM pager based on qt4"
arch=('i686' 'x86_64')
url="http://qutim.org"
license=('unknown')
groups=()
depends=('qt' 'libidn' 'qca')
makedepends=('qt' 'cmake>=2.6' 'qca')
optdepends=()
provides=('qutim')
conflicts=()
replaces=('qutim')
backup=()
options=()
install=
source=()
noextract=()
md5sums=() #generate with 'makepkg -g'
build() {
	mkdir "$srcdir/$pkgname-$pkgver-git"
  cd "$srcdir/$pkgname-$pkgver-git"
	
	git clone git://gitorious.org/qutim/qutim-meta.git
	cd qutim-meta
	git submodule update --init
	rm -rf protocols/quetzal #voodoo compilation errors
	git submodule foreach 'git submodule update --init'
	cmake .
	cmake . # cmake two times - it's important!
	
	make
	
}
