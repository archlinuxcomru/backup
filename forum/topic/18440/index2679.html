display-manager.service
[Unit]
Description=GNOME Display Manager

# replaces the getty
Conflicts=getty@tty1.service
After=getty@tty1.service

# replaces plymouth-quit since it quits plymouth on its own
Conflicts=
After=

# Needs all the dependencies of the services it's replacing
# pulled from getty@.service and
# (except for plymouth-quit-wait.service since it waits until
# plymouth is quit, which we do)
After=rc-local.service plymouth-start.service systemd-user-sessions.service

# GDM takes responsibility for stopping plymouth, so if it fails
# for any reason, make sure plymouth still stops
OnFailure=plymouth-quit.service

[Service]
ExecStart=/usr/bin/gdm
KillMode=mixed
Restart=always
IgnoreSIGPIPE=no
BusName=org.gnome.DisplayManager
StandardOutput=syslog
StandardError=inherit
EnvironmentFile=-/etc/locale.conf
ExecReload=/bin/kill -SIGHUP $MAINPID

[Install]
Alias=display-manager.service

Для этого
sudo systemctl disable gdm

sudo systemctl enable lightdm
Некоторые сервисы командой disable не отключить, связанные сервисы могут запускать их снова. Возможно стоит попробовать отключить через mask. Кроме того, 100% останутся линки gdm, в Centos я удалял два (Они остались после удаления gdm):
rm -f /usr/share/xsessions/*gnome*.desktop
rm -f /usr/share/xsessions/*custom*.desktop
И делал новый для lightdm:
ln -sf /usr/lib/systemd/system/lightdm.service /etc/systemd/system/display-manager.service
