Вам достаточно загрузиться с установщика Арча, а потом как-то так:
mount /dev/sda1 /mnt
mount --bind /dev /mnt/dev
mount --bind /sys /mnt/sys
mount --bind /proc /mnt/proc
pacman -r /mnt -R grub-bios
pacman -r /mnt -S grub-efi-x86_64 (или grub-efi-i386)
chroot /mnt bash
grub-install /dev/sda
grub-mkconfig -o /boot/gub/grub.cfg
