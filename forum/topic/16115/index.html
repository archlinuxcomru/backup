да,приходится монтировать proc и другие ,хотя в
вики написано неправильно,что если

   Если установлен bash 4 или новее, то:
Войдём в chroot

  # /tmp/root.*/bin/arch-chroot /tmp/root.*/

    Иначе:

  # cd /tmp/root.*
  # cp /etc/resolv.conf etc
  # mount --rbind /proc proc
  # mount --rbind /sys sys
  # mount --rbind /dev dev
  # mount --rbind /run run
    (при условии, что /run существует)
  # chroot /tmp/root.* /bin/bash
В общем систему установил,сейчас обобщаю и продумываю минимально необходимый список команд для установки.

Всё,разобрался с установкой.Вот рабочий список команд,может кому понадобиться.
wget http://mirrors.nix.org.ua/linux/archlinux/iso/2016.05.01/archlinux-bootstrap-2016.05.01-i686.tar.gz
sudo mount /dev/sda1 /mnt
cd /mnt
sudo tar -xvf ~/archlinux-bootstrap-2016.05.01-i686.tar.gz
sudo mv /mnt/root.i686/* /mnt
sudo nano /mnt/etc/pacman.d/mirrorlist
sudo -i
#cd /mnt
# cp /etc/resolv.conf etc
# mount --rbind /proc proc
# mount --rbind /dev dev
# chroot /mnt /bin/bash
/]# pacman-key --init
/]# pacman-key --populate archlinux
/]# pacman -Syu
/]# pacman -S base base-devel
в другом  терминале хост-системы sudo swapoff --all
/]# swapon /dev/sda5
/]# genfstab -p / > /etc/fstab
/]# nano /etc/hostname
/]# nano /etc/locale.gen
/]# locale-gen
/]# nano /etc/locale.conf
/]# passwd
/]# ln -s /usr/share/zoneinfo/Europe/Kiev /etc/localtime
/]# useradd -m -g users -G wheel USER
/]# passwd USER
/]# EDITOR=nano visudo
/]# systemctl enable dhcpcd@enp4s0.service
/]# mkinitcpio -p linux
