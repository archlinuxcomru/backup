Так, вот прописал в fonts.conf и сразу стало читабельно :)
    <match target="pattern" name="family" >
        <test name="family" qual="any" >
            <string>helvetica</string>
        </test>
        <edit mode="assign" name="family" >
            <string>Droid Sans</string>
        </edit>
    </match>
