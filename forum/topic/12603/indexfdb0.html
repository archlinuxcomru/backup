заблокирован
 
su
# найдем пакет в котором этот файл запрятан
pkgfile /etc/profile
# выведет core/filesystem, означает что нужно искать в пакете filesystem
# смотрим как реально называется файл пакета
ls /var/cache/pacman/pkg/filesystem*
# выведет что то вроде /var/cache/pacman/pkg/filesystem-2013.05-2-x86_64.pkg.tar.xz
# распакуем
tar -C / -xf /var/cache/pacman/pkg/filesystem-2013.05-2-x86_64.pkg.tar.xz etc/profile
exit
