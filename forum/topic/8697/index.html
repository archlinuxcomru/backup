пользую zsh уже пару лет
если кому интересно ниже лежит конфиг, правда, без коментов 
bindkey '\e[3~' delete-char # del
bindkey ';5D' backward-word # ctrl+left
bindkey ';5C' forward-word #ctrl+right
HISTFILE=~/.zhistory
SAVEHIST=5000
setopt APPEND_HISTORY
setopt HIST_IGNORE_ALL_DUPS
setopt HIST_IGNORE_SPACE
setopt AUTO_CD
zstyle ':completion:*' menu yes select
autoload -U compinit
compinit -C
autoload colors && colors
if [[ $EUID == 0 ]]
then
	PROMPT=$'%{\e[1;31m%}%n %{\e[1;34m%}%~ #%{\e[0m%} ' # user dir %
else
	PROMPT=$'%{\e[1;32m%}%n %{\e[1;34m%}%~ %#%{\e[0m%} ' # root dir #
fi
	RPROMPT=$'%{\e[1;34m%}%T%{\e[0m%}' # right prompt with time
#	RPROMPT="%{$fg_bold[red]%}*%{$reset_color%}"T
SPROMPT='zsh: Меняем '\''%R'\'' на '\''%r'\'' ? [Yes/No/Abort/Edit] '
zstyle ':completion:*:*:kill:*:processes' list-colors "=(#b) #([0-9]#)*=$color[cyan]=$color[red]"
	
alias ls='ls --color=auto'
alias grep='grep --color=auto'
