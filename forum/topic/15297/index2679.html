Возможно ли использовать polkit-gnome-authentication-agent-1 вместо gksu в скрипте для Nemo\Nautilus.

#!/bin/bash
# Get the path
if [ -e -n $1 ]; then
        obj="$NEMO_SCRIPT_SELECTED_FILE_PATHS"
else
        path="`echo $NEMO_SCRIPT_CURRENT_URI | cut -d'/' -f3- | sed 's/%20/ /g'`"
        selected="$path/${1##*/}"
fi

#Check if selected object file or folder
if [ -f "$selected" ]; then
        gksu gedit "$selected"
elif [ -d "$selected" ]; then
        gksu nemo "$selected"
fi

exit 0
