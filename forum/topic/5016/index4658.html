Администратор, разберитесь со своим /etc/yaourtrc для начала…
Например:
...
# AUR 
AURCOMMENT=5
#AURDEVELONLY=0
AURSEARCH=1
AURUPGRADE=1
AURVOTE=0
# Build
EXPORT=1           # Export to 1: EXPORTDIR or PKGDEST
                    # 2: pacman cache (as root)
EXPORTSRC=0        # Need EXPORT>0 to be used
EXPORTDIR="/media/multimedia/localrepo/aur"       # If empty, use makepkg's connfiguration (see makepkg.conf)
# Prompt
#NOCONFIRM=0
#UP_NOCONFIRM=0     # No prompt while build upgrades (including -Sbu)
#BUILD_NOCONFIRM=0  # Only prompt for editing files
#EDITFILES=1
#NOENTER=1
# Output
USECOLOR=1
DETAILUPGRADE=1
#SHOWORPHANS=1
TERMINALTITLE=1
# Command
PACMAN="/usr/bin/pacman" # или "/usr/bin/pacman-color" в том случае, если pacman-color установлен.
#DIFFEDITCMD="vimdiff"
