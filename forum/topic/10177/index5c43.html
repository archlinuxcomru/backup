cat /etc/rc.conf
# LOCALIZATION
# ————
HARDWARECLOCK=“localtime”
TIMEZONE=“Europe/Prague”
KEYMAP=“ru”
CONSOLEFONT=“cyr-sun16”
CONSOLEMAP=“”

# mkinitcpio -p linux

Прошу помочь разобраться, или указать что я упустил.
