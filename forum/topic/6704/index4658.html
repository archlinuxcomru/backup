У меня несколько linux машин включены в домен. Так вот, когда смотришь на сервере w2k3 открытые файлы (\pipe\lsarpc и \pipe\netlogon и \pipe\samr), там высвечиваются только эти машины, виндовые машины (тоже в домене) там не высвечиваются. Почему так?

Мой smb.conf 
[global]
netbios name = SMBSERVER
security = ads
server string =
workgroup = MYDOM
realm = MYDOM.LOCAL
os level = 33
domain master = no
local master = no
preferred master = no
dns proxy = no
disable netbios = no
guest account = nobody
auth methods = winbind
password server = 192.168.1.23
encrypt passwords = yes
unix password sync = Yes
passwd program = /usr/bin/passwd %u
passwd chat = *Enter\snew\sUNIX\spassword:* %n\n *Retype\snew\sUNIX\spassword:* %n\n *password\supdated\ssuccessfully* .
passwd chat debug = yes
smb ports = 139
bind interfaces only = true
interfaces = eth0 127.0.0.1/8
hosts deny = all
hosts allow = 192.168.1.0/24 
add user script = /usr/sbin/useradd -m %u -g ntusers -G ntusers
delete user script = /usr/sbin/userdel -r %u
add group script = /usr/sbin/groupadd %g
delete group script = /usr/sbin/groupdel %g
add user to group script = /usr/sbin/usermod -G %g %u
add machine script = /usr/sbin/useradd -N -g machines -c Machine -s /bin/false/ -d /tmp  %u
client use spnego = yes
client ntlmv2 auth = yes
restrict anonymous = 2
allow trusted domains = yes
winbind use default domain = yes
idmap gid = 10000-200000
idmap uid = 10000-200000
winbind uid = 10000-200000
winbind gid = 10000-200000
winbind enum users = yes
winbind enum groups = yes
winbind nested groups = no
winbind refresh tickets = yes
winbind use default domain = yes
#winbind separator = @
acl map full control = true
inherit owner = yes
inherit acls = yes
map acl inherit = yes
inherit permissions = yes
nt acl support = yes 
admin users = root,admin,@MYDOM\\"администраторы домена",MYDOM\\admin
#valid users = "пользователи домена","компьютеры домена","администраторы домена"
passdb backend = tdbsam
template shell = /bin/bash
display charset = UTF-8
unix charset = UTF-8
dos charset = cp866
wide links = no
log level = 0
#+Отключение печати
load printers = no
show add printer wizard = no
printcap name = /dev/null
disable spoolss = yes
#+Отключение печати
[netlogon]
 comment = Network Logon Service
 path = /var/lib/samba/netlogon
 guest ok = yes
 writable = no
[IPC$]
path = /tmp
write list = @MYDOM\\"пользователи домена"
create mask = 755
admin users = root,MYDOM\\admin
[Profiles]
    path = /var/lib/samba/profiles
    browseable = no
    guest ok = yes
[C$]
   comment = Administrative share for homes
   path = /home
   writable = yes

Linux машины в домене работают нормально, авторизация на них идет через контроллер домена посредством winbind.
