Для последнего Manjaro рабочяя конфигурация пункта загрузки с iso в меню grub.
Может кому надо.Образ ложил в папку /boot/iso

         menuentry "Manjaro  grub_iso" --class dvd {
         set isofile="/boot/iso/manjaro-xfce-openrc-17.0.1-i686.iso"
         set dri="free"
         search --no-floppy -f --set=root $isofile
         probe -u $root --set=abc
         set pqr="/dev/disk/by-uuid/$abc"
         loopback loop $isofile
         linux  (loop)/boot/vmlinuz-i686  img_dev=$pqr img_loop=$isofile driver=$dri
         initrd  (loop)/boot/intel_ucode.img (loop)/boot/initramfs-i686.img
        }

В /etc/grub.d/40_custom записано так:

#!/bin/sh
exec tail -n +3 $0
# This file provides an easy way to add custom menu entries.  Simply type the
# menu entries you want to add after this comment.  Be careful not to change
# the 'exec tail' line above.

# Clonezilla Live for GRUB
================================================================================
menuentry "Clonezilla live" {
set isofile="/home/tozen/iso/clonezilla.iso"
loopback loop $isofile
linux (loop)/live/vmlinuz boot=live union=overlay username=user config components quiet noswap nolocales edd=on nomodeset ocs_live_run=\"ocs-live-general\" ocs_live_extra_param=\"\" keyboard-layouts= ocs_live_batch=\"no\" locales= vga=788 ip=frommedia nosplash toram=filesystem.squashfs findiso=$isofile
initrd (loop)/live/initrd.img
}
================================================================================
