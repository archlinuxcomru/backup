Да sed тут и не нужен тогда.
Как-то так должно быть.
#!/bin/bash

readarray -t TEXT < "$1"
for i in "${!TEXT[@]}"; do
	if ! ([ -z "${TEXT[i]}" ] && [ -n "${TEXT[i-1]}" ] && [ -n "${TEXT[i+1]}" ]); then
		echo "${TEXT[i]}"
	fi
done
