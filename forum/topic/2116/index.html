В системнике две сетевухи - одна отвечает за Инет, другая за раздачу его через точку доступа. При перезагрузке Арч нагло меняет их местами и приходится перевтыкать штекеры. Что делать, чтобы Арч больше не шалил?

rc.conf

#
# /etc/rc.conf - Main Configuration for Arch Linux
#
# See 'man 5 rc.conf' for more details
#
# LOCALIZATION
# ------------
HARDWARECLOCK="localtime"
TIMEZONE="Asia/Yekaterinburg"
KEYMAP="ru"
CONSOLEFONT="ter-u16b"
CONSOLEMAP=""
LOCALE="ru_RU.UTF-8"
DAEMON_LOCALE="yes"
USECOLOR="yes"
# HARDWARE
# --------
MODULES=(floppy 8139too 8139cp)
USEDMRAID="no"
USEBTRFS="no"
USELVM="no"
# NETWORKING
# ----------
HOSTNAME="localhost"
lan="eth0 192.168.1.2 netmask 255.255.255.0 broadcast 192.168.1.255"
wlan="eth1 192.168.0.1 netmask 255.255.255.0 broadcast 192.168.0.255"
INTERFACES=(lan wlan)
gateway="default gw 192.168.1.1"
ROUTES=(gateway)
#interface=
#address=
#netmask=
#broadcast=
#gateway=
NETWORK_PERSIST="no"
# DAEMONS
# -------
#
DAEMONS=(hwclock syslog-ng network hal dbus netfs iptables crond fam alsa ntpd kdm pacmanbd)

10-network.rules

SUBSYSTEM=="net", ATTRS{address}=="00:1e:8c:0e:c0:fc", NAME=="lan"
SUBSYSTEM=="net", ATTRS{address}=="4c:00:10:61:0d:23", NAME=="wlan"
