[Unit]
Description=dncpcd on %I
Wants=network.target
Before=network.target
BindsTo=sys-subsystem-net-devices-%I.device

[Service]
type=forking
PidFile=/run/dncpcd-%I.pid
ExecStart=/usr/sbin/dhcpcd -1 -w %I
ExecStop=/usr/sbin/dncpcd -x %I

[Install]
Alias=multi-user.target.wants/[email protected]
