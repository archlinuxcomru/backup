aur/oblogout - шрифты не настраиваются, или я не увидел как и где. locale/ru нет. Зато есть любимый многими Python.
aur/ob-pippe-menu (w00zy) - в PKGBUILD'е ссылки вникуда (полтора года не обновлялись, или я опять что-то интересное проспал), поправил PKGBUILD, собрал oblogout-0.2.6-1-i686.pkg.tar.xz
Будет настроение, гляньте:
PKGBUILD
pkgname=oblogout
pkgver=0.2.6
pkgrel=1
pkgdesc="Openbox pippe menu for XDG system menu"
arch=('i686' 'x86_64')
license=('GPL2')
url="http://downloads.sourceforge.net/project/staybox"
groups=()
depends=('gtk2' 'glibc' 'gnome-menus')
makedepends=('pkgconfig')
source=(http://sourceforge.net/projects/staybox/files/${pkgname}-${pkgver}.tar.gz/download)
build() {
    cd ${srcdir}/${pkgname}-${pkgver}
    ./configure --sysconfdir=/etc --prefix=/usr || return 1
    make || return 1
    make DESTDIR=${pkgdir} install || return 1
}
md5sums=('3864a2488037f3dea5d20623b8f520bf')
