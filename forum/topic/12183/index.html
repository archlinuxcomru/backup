local function dispcpu()
	local f, infos
	local capi = {
		mouse = mouse,
		screen = screen
	}
	f = io.popen("echo '%CPU   PID COMMAND' && ps hgaxo %cpu,pid,comm | sort -nrk1 | head -n 10 | sed -e 's/-bin//;s/-media-play//'" )
	infos = f:read("*all")
	f:close()
	showmeminfo = naughty.notify( {
	text	      = infos,
	font          = font,
        fg            = beautiful.fg_blue,
        bg            = beautiful.fg_black,
        timeout	      = 0,
        hover_timeout = 0.5,
        position      = "top_right",
        border_color  = beautiful.border_tooltip,
        border_width  = 1,
		screen	      = capi.mouse.screen })
end
cpuwidget:connect_signal('mouse::enter', function () dispcpu(path) end)
