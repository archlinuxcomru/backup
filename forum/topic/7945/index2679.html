предлагаю собрать в одном месте что можно сделать для облегчения и ускорения арчи, для затравки - вот такой скрипт слизанный с основного форума
#!/bin/bash
### Bind temporary directories to /dev/shm ###
# I do this instead of mounting tmpfs on the #
# directories, so less memory gets wasted.   #
##############################################
mkdir /dev/shm/{tmp,lock,pkg}
mount --bind /dev/shm/tmp /tmp
mount --bind /dev/shm/tmp /var/tmp
mount --bind /dev/shm/lock /var/lock
mount --bind /dev/shm/pkg /var/cache/pacman/pkg
chmod 1777 /dev/shm/{tmp,lock,pkg}
