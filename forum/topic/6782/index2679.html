Я настраивал перловый скрипт. У меня так работает:
location ~ \.cgi$ {
        root /srv/http/cgi-bin;
        fastcgi_pass   localhost:9001;
        fastcgi_index  index.php;
        fastcgi_param  SCRIPT_FILENAME  $fastcgi_script_name;
        fastcgi_param  QUERY_STRING     $query_string;
        fastcgi_param  REQUEST_METHOD   $request_method;
        fastcgi_param  CONTENT_TYPE     $content_type;
        fastcgi_param  CONTENT_LENGTH   $content_length;
        include        fastcgi_params;
Поменяй SCRIPT_FILENAME и добавь остальные.

Вобщем я победил. Посидел, покурил, подумал и усе заработало. Для этого пришлось еще собрать пакет fcgiwrap
Вот часть конфига отвечающего за стату:
location /awstats {
        root /srv/http/awstats/cgi-bin;
        fastcgi_pass   localhost:9001;
        fastcgi_index  index.php;
        fastcgi_param  SCRIPT_NAME      /awstats.pl;
        fastcgi_param  QUERY_STRING     $query_string;
        fastcgi_param  REQUEST_METHOD   $request_method;
        fastcgi_param  CONTENT_TYPE     $content_type;
        fastcgi_param  CONTENT_LENGTH   $content_length;
        include        fastcgi_params;        
      }      
        location /awstatsclasses/ {
        alias /srv/http/awstats/classes/;
        }
        location /awstatscss/ {
        alias /srv/http/awstats/css/;
        }
        location /awstatsicons/ {
        alias /srv/http/awstats/icon/;
        }
