.xinitrc

#!/bin/sh
if [ -d /etc/X11/xinit/xinitrc.d ]; then
  for f in /etc/X11/xinit/xinitrc.d/*; do
    [ -x "$f" ] && . "$f"
  done
  unset f
fi
setxkbmap -layout us,ru -variant ,winkeys -option grp:ctrl_shift_toggle,grp_led:scroll,terminate:ctrl_alt_bksp,compose:ralt
exec xterm
