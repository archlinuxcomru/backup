Попробуй на кошках сначала
find -type f -name "default.htm"
./default.htm
./1/default.htm
./1/2/default.htm
./1/2/3/default.htm
./1/2/3/4/default.htm

find -type f -name "default.htm" -print0 | xargs -0 perl-rename 's/default/index/'
find -type f -name "index.htm"
./index.htm

./1/2/index.htm
