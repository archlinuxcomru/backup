UPD. Не дали обновить устаревший fbxkb. Простой индикатор раскладок. Работает на ура.
PKGBUILD
pkgname=fbxkb
pkgver=0.6
pkgrel=1
pkgdesc="Keyboard indicator and switcher"
url="http://fbxkb.sourceforge.net/"
depends=('gtk2')
license="GPL"
source=(http://switch.dl.sourceforge.net/sourceforge/fbxkb/${pkgname}-${pkgver}.tgz
fbxkb.desktop)
arch=('i686')
build() {
cd $startdir/src/${pkgname}-$pkgver
# Disable Debug, Optimize CFLAGS
sed -i "s|-g |$CFLAGS |g" Makefile.common
sed -i "s|-O2|$CFLAGS|g" Makefile.common
./configure --prefix=/usr
make || return 1
make PREFIX=$startdir/pkg/usr install || return 1
cp $startdir/src/${pkgname}-${pkgver}/images/us.png $startdir/pkg/usr/share/fbxkb/images/zz.png || return 1
mkdir -p $startdir/pkg/usr/share/applications/
install -D -m 644 ${startdir}/src/fbxkb.desktop $startdir/pkg/usr/share/applications/fbxkb.desktop || return 1
}
md5sums=('fa768bbb07aac8a4ae590633499cce15')

fbxkb.desktop
[Desktop Entry]
Version=1.0
Encoding=UTF-8
Type=Application
Terminal=false
Name=FBXKB
Exec=fbxkb
Comment=Simple KBD indicator
Comment[ru_RU]=Простой индикатор раскладок
GenericName=Simple KBD indicator
GenericName[ru]=Индикатор раскладок
Icon=preferences-desktop-keyboard
Categories=Application;System;Utility;
