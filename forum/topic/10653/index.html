Все заработало.
так как решение было

Edit this file: /usr/share/polkit-1/actions/org.freedesktop.udisks.policy and look for the part with id org.freedesktop.udisks.filesystem-mount, proceed to the tag defaults. This says:

<defaults>
<allow_any>no</allow_any>
<allow_inactive>no</allow_inactive>
<allow_active>yes</allow_active>
</defaults>
and change this to
<defaults>
<allow_any>yes</allow_any>
<allow_inactive>yes</allow_inactive>
<allow_active>yes</allow_active>
</defaults>

то подправил вариант из вики
// Allow udisks2 to mount devices without authentication
// for users in the "storage" group.
polkit.addRule(function(action, subject) {
 if ((action.id == "org.freedesktop.udisks2.filesystem-mount-system" ||
      action.id == "org.freedesktop.udisks2.filesystem-mount") && subject.isInGroup("storage")) {
       return polkit.Result.YES;
   }
});
polkit.addRule(function(action, subject) {
   if ((action.id == "org.freedesktop.udisks.filesystem-mount" ||       // <----- добавил эту строчку
        action.id == "org.freedesktop.udisks.filesystem-mount-system-internal") && subject.isInGroup("storage")) {
       return polkit.Result.YES;
    }
});
и теперь все монтируется и через дельфин и через виджет, всем спасибо за помощь.
