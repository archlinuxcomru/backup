...
Section "Monitor"
...
    Modeline "1152x864_100.00"  143.47  1152 1232 1360 1568  864 865 868 915  -HSync +Vsync
EndSection
...
Section "Screen"
...
    SubSection     "Display"
        Depth       24
	Viewport   0 0
	Modes       "1152x864_100.00"
    EndSubSection
EndSection

$ gtf 1152 864 100
