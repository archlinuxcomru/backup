Здравствуйте! Хочу настроить у себя на компьютере VPN в рабочую сеть, создал себе на микротике учетку l2tp, начал настраивать strongswan и уперся в стену -
Конфиг

# ipsec.conf - strongSwan IPsec configuration file

# basic configuration

config setup
        # strictcrlpolicy=yes
        # uniqueids = no

# Add connections here.

# Sample VPN connections
conn work
    auto=start
    right=внешний_адрес
    rightid=mikrotik
    rightsubnet=0.0.0.0/0
    rightauth=secret
    rightprotoport=17/1701
    #leftsourceip=%config
    leftid=tvv
    leftauth=eap-mschapv2
    eap_identity=%identity
