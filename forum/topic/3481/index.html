вот собственно вопрос: в smb.conf установлен режим безопасности share, есть шары гостевые, а есть шары которые я сделал не анонимными, но если зайти на общую шару и кликнуть по не анонимной шаре, то каталог пытается открыться некоторое время, а потом появляется окно с “отказано в доступе”. Если например подключить шару через “подключить сетевой диск” и ввести указанный логин как в smb.conf с паролем, то шара подключается. С анонимными шарами проблем нет. А вот собственно я хочу сделать так чтоб юзеру кликающиему по запороленной шаре предлагалось ввести логин/пароль и после успешного ввода он попадал в каталог собственно, а на анонимных шарах всё так и осталось как есть, причём security = user не предлагать, так как она требует ввода логина пароля ещё до захода на общую шару. Вот собственно конфиг
smiges 20:49 [0]~$ cat /etc/samba/smb.conf | grep -Ev '^#|#|^$'
[global]
	workgroup = Samba
	netbios name = smiges
	server string = Arch Linux Server
	interfaces = eth0, lo
	security = SHARE
	guest account = qwerty
	client lanman auth = Yes
	log file = /var/log/samba/%I.log
	max log size = 2048
	socket options = TCP_NODELAY SO_RCVBUF=8192 SO_SNDBUF=8192
	load printers = No
	local master = YES
	domain master = No
	dns proxy = No
	hosts allow = 192.168.
[share]
	path = /home/smiges/media/samba/

	guest ok = Yes
