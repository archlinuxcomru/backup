Если, установку системы, начинали с # loadkeys ru , то вся система русифицирована :
 beisic ~ $  cat /etc/locale.conf
LANG=ru_RU.UTF-8
Или, не сделали настройку файла ~.bashrc, например :
 beisic ~ $  cat .bashrc
#
# ~/.bashrc
#

# If not running interactively, don't do anything
[[ $- != *i* ]] && return

export HISTCONTROL=ignoreboth
export HISTIGNORE='history*'
export PROMPT_COMMAND='history -a;echo -en "\e]2;";history 1|sed "s/^[ \t]*[0-9]\{1,\}  //g";echo -en "\e\\";'
