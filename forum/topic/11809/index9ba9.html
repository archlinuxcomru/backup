опенбокс не взлетел
Чего это не взлетел?

Вот пример без слима, обычный вход и консоле и автозапуск опенбокса в первой консоле, ну или чего нибудь другого.
Что тут сложного? не понимаю.
~/.xinitrc
#!/bin/sh
#
# ~/.xinitrc
#
# Executed by startx (run your window manager from here)
if [ -d /etc/X11/xinit/xinitrc.d ]; then
  for f in /etc/X11/xinit/xinitrc.d/*; do
    [ -x "$f" ] && . "$f"
  done
  unset f
fi
# exec gnome-session
# exec startkde
# exec startxfce4
# ...or the Window Manager of your choice
exec openbox-session

и в ~/.bash_profile или в ~/.zprofile
дописать
[[ -z ${DISPLAY} && ${XDG_VTNR} -eq 1 ]] && exec startx &> ~/.xlog

А сюда ~/.config/openbox/autostart прописываются команды которые нужно запускать при запуске опенбокса
Вот мой пример
#
# These things are run when an Openbox X Session is started.
# You may place a similar script in $HOME/.config/openbox/autostart
# to run user-specific things.
#
# If you want to use GNOME config tools...
#
#if test -x /usr/lib/openbox/gnome-settings-daemon >/dev/null; then
#  /usr/lib/openbox/gnome-settings-daemon &
#elif which gnome-settings-daemon >/dev/null 2>&1; then
#  gnome-settings-daemon &
#fi
# If you want to use XFCE config tools...
#
#xfce-mcs-manager &
[[ -f ~/.Xresources ]] && xrdb -merge ~/.Xresources &
[[ -f ~/.Xmodmap ]] && xmodmap ~/.Xmodmap &
/usr/lib/notify-osd/notify-osd &
nitrogen --restore &
tint2 &
volumeicon &
numlockx &
sleep 2 && conky -c ~/.config/conky/conkyrc1 -q &
sleep 3 && conky -c ~/.config/conky/conkyrc2 -q &
sleep 4 && conky -c ~/.config/conky/conkyrc3 -q &
sleep 5 && sbxkb &
#xscreensaver -no-splash &
