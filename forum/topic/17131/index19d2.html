#!/usr/bin/env zsh

разминимизировать :)
#!/usr/bin/env zsh
MW1=/tmp/$USER/MW
MW2=/tmp/$USER/minwin
F1=`cat $MW1`
F2=$(( $F1 - 1 ))
if [[ $F2 -eq 1 ]]; then
    W="window"
else W="windows"
fi
if [[ $F1 -gt 1 ]]
    then echo $F2 > $MW1 && echo '<fc=#ff6500><fn=1>'$F2'</fn></fc>' $W > $MW2

fi
