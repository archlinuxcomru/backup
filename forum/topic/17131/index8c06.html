mkfifo /tmp/MW

    , ((modm,                            0x7a),  withFocused minimizeWindow <+> spawn "MW2")                              --Mod4+Z
    , ((modm,                            0x61),  sendMessage RestoreNextMinimizedWin <+> spawn "MW3")                     --Mod4+A
MW2
#!/usr/bin/env zsh
echo "True" > /tmp/MW
exit 0

#!/usr/bin/env zsh
echo "False" > /tmp/MW
exit 0
xmonarrc
