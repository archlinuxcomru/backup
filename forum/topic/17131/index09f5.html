#!/bin/sh
 if (pidof dunst >/dev/null); then kill $(pidof dunst)
 else curl -H "Accept-Language: ru" wttr.in/Mednogorsk?T?q?n?0 | xargs -0 notify-send -t -0
 fi
 exit 0
?T - терминальный режим, чтобы в dunst работало, без этого у меня кракозябры))
?q - сокращенный вывод города
?n - вывод только день/ночь
?0 - текущая погода (1, 2, 3- 1, 2, 3 - дней соответственно)
