на первом точно, вот
ubuntu@ubuntu:~$ sudo mount /dev/sda1 /mnt
ubuntu@ubuntu:~$ 

$ sudo ln -s usr/bin /mnt/sbin
ubuntu@ubuntu:~$ sudo ln -s usr/bin /mnt/bin
ubuntu@ubuntu:~$ sudo ln -s bin /mnt/usr/sbin
ubuntu@ubuntu:~$
