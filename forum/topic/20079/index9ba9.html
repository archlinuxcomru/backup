проще
Загрузиться с загрузочного исо арча, поднять сеть
примонтировать корневой раздел в /mnt, бут не монтировать.
и 
arch-chroot /mnt
pacman -Syu  linux-headers linux linux-lts-headers linux-lts os-prober grub
#если стоял блоб нвидии то еще pacman -S nvidia-lts, а если dkms модули переставить и их
nano /mnt/etc/fstab
# закомментируйте монтирование раздела в бут.
#проверьте что корневой раздел прописан с указанием uuid а не /dev
смонтируйте раздел бут в /media
и
grub-mkconfig -o /media/grub/grub.cfg
exit
reboot

при загрузке в меню груба должны быть пункты загрузки с обычным и лтс ядром
