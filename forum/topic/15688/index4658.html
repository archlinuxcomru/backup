То есть, нужен однородный, без "доп.эффектов" скроллбар.
В данный момент используется такой userChrome.css:
menubar > menu { color: #ffffff !important; } menubar > menu[_moz-menuactive="true"][open="false"] { color: #ffffff !important; } menubar > menu[_moz-menuactive="true"][open="true"] { color: #ffffff !important; } /* MenuColors, Added by QtCurve -- do not remove */

/* Toolbar panel, page, findbar */

toolbar
{
  background: #F5F6F7 !important;
}

page
{
  overflow: -moz-hidden-unscrollable;
  -moz-box-orient: vertical;
  background: #F5F6F7 !important;
}

sidebarheader {
  -moz-appearance: none;
  height: 32px;
  background: #F5F6F7 !important;
}

scrollbar {
  -moz-appearance: none !important;
  background: #F2F3F3 !important;
}


scrollbar > scrollbarbutton { 
  -moz-appearance: none !important; 
  border: none !important; 
  background: none !important; 
}

scrollbar[orient='vertical'] > scrollbarbutton { 
  min-height: 0 !important; 
}

scrollbar[orient='horizontal'] > scrollbarbutton {
  min-width: 0 !important;
}
