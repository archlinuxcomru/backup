Не могу решить проблему с зависимостями. Собирается, но не стартует. Не знаю, что надо туда добавить
pkgname=awn-applet-appmenu
pkgver=0.0.3
pkgrel=1
pkgdesc="Avant window navigator menu applet from Zorin OS"
url="https://launchpad.net/~zorin-os"
arch=('x86_64' 'i686')
depends=('libindicator-gtk3')
optdepends=()
makedepends=()
conflicts=()
replaces=()
backup=()
source=("https://launchpad.net/~zorin-os/+archive/packages/+files/awn-applet-appmenu_0.0.3.tar.gz")
md5sums=(SKIP)
build() {
  cd "${srcdir}/${pkgname}"
  make
}
package() {
  cd "${srcdir}/${pkgname}"
  make DESTDIR="${pkgdir}" install
  install -Dm644 ./debian/copyright "$pkgdir/usr/share/licenses/$pkgname/COPYING"
}
