Auto Reconnect = on
Init2 = AT+CGDCONT=1,“IP”,“internet.mts.ru”
Baud = 115200
Init1 = ATZ
Modem = /dev/ttyACM0
Phone = *99#
Username = mts
Abort on Busy = on
Password = mts
Stupid Mode = on

#wvdial zte
