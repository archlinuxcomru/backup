set_proxy ()
{
PROXY=(`wget -qO- "http://proxy-list.org/english/search.php?search=&country=any&type=elite&port=any" | grep -v "*" | grep '<li class="proxy">[^<]*</li>' | sed s/[^0-9.:]//g`)
sed -i '/proxy/d' $HOME/.bashrc
echo -e "export https_proxy="https://${PROXY[1]}"\nexport http_proxy="http://${PROXY[1]}"\nexport ftp_proxy="ftp://${PROXY[1]}"">> $HOME/.bashrc
echo "${PROXY[1]}"
}

unset_proxy ()
{
sed -i '/proxy/d' $HOME/.bashrc
echo "proxy off"
}

while getopts ":su" option ;
do
    case $option in
        s) set_proxy
	;;
        u) unset_proxy
	;;
        *) echo "-s set_proxy -u unset_proxy ";
            exit 1
	;;
        esac
done
