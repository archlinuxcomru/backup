Делал PKGBUILD-ы для себя. Не знаю только как логин и пароль от mercurial в PKGBUILD засунуть.
pkgname=xneur-mercurial
pkgver=0.13
pkgrel=1
pkgdesc="X neural layout switcher - it detects the language of input and corrects it if needed (mercurial-version)"
url="http://xneur.ru/"
arch=(i686 x86_64)
license="GPL"
depends=()
makedepends=('hgsvn')
optdepends=()
provides=('xneur')
conflicts=('xneur')
source=()
md5sums=()
install=xneur-mercurial.install
build() {
	cd $startdir/src/    
	hg clone http://hg.itways.ru/xneur
	cd $startdir/src/xneur/xneur
	./autogen.sh --prefix=/usr --sysconfdir=/etc --without-xosd --without-keylogger
	make || return 1
   	make DESTDIR=$startdir/pkg install
}

xneur-mercurial.install
post_install() {
  gtk-update-icon-cache -q /usr/share/icons/hicolor
}
post_upgrade() {
  post_install
}
post_remove() {
  post_install
}
