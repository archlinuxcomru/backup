дравишки ставил так, долго мучался с конфликтами между пакетиками, может кому полезно будет:
для начала удаляем новые пакетики:
[[email protected] ~]$ sudo pacman -Rd xorg-server
(1/1) удаление xorg-server                          [#####################] 100%
[[email protected] ~]$ sudo pacman -Rd xf86-input-evdev
(1/1) удаление xf86-input-evdev                     [#####################] 100%
[[email protected] ~]$ sudo pacman -Rd xf86-video-ati
(1/1) удаление xf86-video-ati                       [#####################] 100%
[[email protected] ~]$ sudo pacman -Rd xf86-video-vesa
(1/1) удаление xf86-video-vesa                      [#####################] 100%
затем устанавливаю старенькие версии всех выше удаленных пакетиков: 
[[email protected] ~]$ sudo[i] pacman -U[/i] /home/psl-arch/Downloads/catalyst/xf86-video-vesa-2.2.0-1-i686.pkg.tar.gz 
чтение информации о пакете...
проверка зависимостей...
(1/1) проверка возможных конфликтов файлов          [#####################] 100%
(1/1) установка xf86-video-vesa                     [#####################] 100%
и так все 4…
Затем ставлю каталист
[[email protected] ~]$ sudo pacman -U /home/psl/catalyst-10.1-4-i686.pkg.tar.gz 
чтение информации о пакете...
проверка зависимостей...
ошибка: замещение пакетов с использованием параметров -A и -U еще не реализовано
ошибка: вы можете заменить пакеты вручную, используя -Rd и -U
ошибка: не удалось подготовить запрос (конфликтующие зависимости)
:: catalyst: конфликтует с libgl
:: catalyst: конфликтует с xf86-video-ati
:: catalyst: конфликтует с ati-dri
Удаляю все что конфликтует
[[email protected] ~]$ sudo pacman -Rd libgl xf86-video-ati ati-dri
(1/3) удаление xf86-video-ati                       [#####################] 100%
(2/3) удаление ati-dri                              [#####################] 100%
(3/3) удаление libgl                                [#####################] 100%
Затем ставлю катлист :
[[email protected] ~]$ sudo pacman -U /home/psl/catalyst-10.1-4-i686.pkg.tar.gz 
чтение информации о пакете...
проверка зависимостей...
(1/1) проверка возможных конфликтов файлов          [#####################] 100%
(1/1) установка catalyst                            [#####################] 100%
--------------------------------------------------------------
You can use the tool 'aticonfig' to generate an xorg.conf file.
Remember to add fglrx to the MODULES list in /etc/rc.conf.
--------------------------------------------------------------
Просит добавить в модули, добавляю…
"MODULES=(fglrx)"
затем правим ручками /etc/X11/xorg.conf , если его нет то создаем командой Xorg -configure.
Вот так выгляюедт мой xorg.conf
Section "ServerLayout"
	Identifier     "X.org Configured"
	Screen      0  "Screen0" 0 0
	InputDevice    "Mouse0" "CorePointer"
	InputDevice    "Keyboard0" "CoreKeyboard"
EndSection
Section "Files"
        RgbPath      "/usr/share/X11/rgb"
	ModulePath   "/usr/lib/xorg/modules"
	FontPath     "/usr/share/fonts/misc"
	FontPath     "/usr/share/fonts/100dpi:unscaled"
	FontPath     "/usr/share/fonts/75dpi:unscaled"
	FontPath     "/usr/share/fonts/TTF"
	FontPath     "/usr/share/fonts/Type1"
EndSection
Section "Module"
        Load  "extmod"
        Load  "xtrap"
        Load  "record"
	Load  "dri2"
	Load  "dbe"
	Load  "extmod"
	Load  "dri"
	Load  "glx"
        Load  "GLcore"
        Load  "freetype"
EndSection
Section "InputDevice"
	Identifier  "Keyboard0"
	Driver      "kbd"
EndSection
Section "InputDevice"
	Identifier  "Mouse0"
	Driver      "mouse"
	Option	    "Protocol" "auto"
	Option	    "Device" "/dev/input/mice"
	Option	    "ZAxisMapping" "4 5 6 7"
EndSection
Section "Monitor"
	#DisplaySize	  340   190	# mm
	Identifier   "Monitor0"
	VendorName   "SEC"
	ModelName    "3251"
EndSection

EndSection

        Device     "Card0"

        DefaultDepth    24
        SubSection "Display"
                Viewport   0 0
                Depth     24
        EndSubSection
EndSection
   Section "DRI"
       Mode 0666
   EndSection
