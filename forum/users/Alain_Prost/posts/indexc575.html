acestreamengine --client-gtk --bind-all --login LOGIN --password PASSWORD --log-file $ACE_LOG 2>&1 &

Если потом будут проблемы c libappindicator, установите из ayatana (добавить эти строки в /etc/pacman.conf) :

Server = http://pkgbuild.com/~bgyorgy/$repo/os/$arch

sudo pacman -Sy
sudo pacman -S libappindicator
