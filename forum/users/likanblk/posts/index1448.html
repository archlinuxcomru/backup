Мой провайдер предоставляет определённые конфиги для настройки Xl2tpd, в частности
отредактировать файл /etc/network/interfaces

auto lo
iface lo inet loopback
address 127.0.0.1
netmask 255.0.0.0

#Ниже подразумевается, что внешний интерфейс соединённый с локальной сетью у вас eth0
#если это не так замените его на необходимый
auto eth0
iface eth0 inet static
address ваш_IP
netmask 255.255.255.0
gateway ваш_шлюз
up route add -net 192.168.0.0 netmask 255.255.0.0 gw ваш_шлюз dev eth0

Но в арче файл отсутствует, куда вбивать соответствующие строчки?
