/etc/inittab:
Заменить
id:3:initdefault:
[...]
x:5:respawn:/usr/X11R6/bin/xdm -nodaemon
на
id:5:initdefault:
[...]
x:5:once:/bin/su <USER> -l -c "/bin/bash --login -c startx >/dev/null 2>/dev/null"
http://wiki.archlinux.org/index.php/Start_X_at_Boot
