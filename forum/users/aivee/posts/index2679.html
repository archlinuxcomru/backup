url="https://launchpad.net/cairo-dock-core"
license="GPL"

options=('!libtool')
source=(http://launchpad.net/$pkgname-core/2.1/$pkgver/+download/$pkgname-$pkgver-$subversion.tar.gz)
build() {
  cd $srcdir/$pkgname-$pkgver-$subversion
  autoreconf -isvf ||return 1
  ./configure --prefix=/usr --disable-static ||return 1
  make ||return 1
  make install DESTDIR=$pkgdir ||return 1
}
