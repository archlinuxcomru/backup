Содержимое:
$ cat /lib/systemd/system-shutdown/debug.sh
#!/bin/sh
mount -o remount,rw /
dmesg -T > /var/log/shutdown.log
echo '/var/log/shutdown.log saved!!!'
mount -o remount,ro /
