172.16.0.0      10.0.5.2        255.255.248.0   UG    1      0        0 eth1
192.168.128.0   10.0.5.2        255.255.224.0   UG    1      0        0 eth1
10.0.0.0        *               255.255.192.0   U     0      0        0 eth1
default         10.0.3.2     0.0.0.0         UG    3      0        0 eth1

resolv.conf
nameserver 192.168.128.254
domain ionitcom.ru
domain jagry.ru
nameserver 192.168.128.251

domain jagry.ru
nameserver 10.0.3.2
