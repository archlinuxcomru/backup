Сделал всё (почти) один в один, как товарищ свыше. А именно:

/etc/network.d/corbina
CONNECTION='ethernet'
DESCRIPTION='Beeline'
INTERFACE='eth0'
IP='dhcp'

/etc/rc.conf
NETWORKS=(corbina)
POST_UP="ip route add 213.234.192.8 dev eth0 ; ip route add 85.21.192.3 dev eth0 ; ip route add 85.21.0.251 via 10.163.88.1 dev eth0 metric 202"

/etc/xl2tpd/xl2tpd.conf
[global]
access control = yes
auth file = /etc/ppp/chap-secrets
[lac corbina]
lns = 85.21.0.251
redial = yes
redial timeout = 5
require chap = yes
require authentication = no
name = 0893304887
ppp debug = yes
pppoptfile = /etc/ppp/options.xl2tpd
require pap = no
autodial = yes

/etc/ppp/options.xl2tpd
name 0893304887
remotename L2TP
ipparam corbina
connect /bin/true
mru 1460
mtu 1460
nodeflate
nobsdcomp
persist
maxfail 0
nopcomp
noaccomp
defaultroute
usepeerdns
