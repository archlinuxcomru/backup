pkgver=0.0.3
pkgrel=1
pkgdesc="Avant window navigator menu applet from Zorin OS"
url="https://launchpad.net/~zorin-os"
arch=('x86_64' 'i686')
depends=('libindicator-gtk3')
optdepends=()
