От того что мы закинем все хомяки ситуация не изменится, так как вы говорили что у Вас один юзер. Т.е. по факту “хрен редьки не слаще”, но геморя меньше.
;; 
# mount /dev/sda3 /mnt/temp
; Сносим все
# rm -R /mnt/temp/*
; Копируем хомяки 
# cp -a /home/* /mnt/temp
; на всякий
# chown -R mazahaka:users /mnt/temp/mazahaka
; Отрываем
# umount /mnt/temp

/dev/sda3 /home ext4 defaults 0 0
