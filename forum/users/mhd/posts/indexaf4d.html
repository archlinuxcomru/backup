Имеется PKGBUILD:


pkgname=grabtor
pkgver=0.1
pkgrel=1
pkgdesc="A graphic battery status monitor adapted for xmobar."
arch=('any')
license=('GPL3')
makedepends=('ghc')
source=('GraphicBatteryMonitor.hs' 'Main.hs' 'Setup.hs' 'grabtor.cabal' 'LICENSE')
md5sums=('cdacd35d118d86d6038dd0903931cc77'
         '5184592635022ce81fc9b0cfedfaf4c7'
         'bbcbe0b92a3eaebee9d6d76f3b87aa95'
         'e76e0037cfff9ffc2b921609cb28c77b'
         '779db0e1e5d7f9b9c97d22e2a2f20123')

build() {
	runhaskell Setup configure --prefix=/usr --ghc-options=-dynamic
	runhaskell Setup build
}

package() {
	runhaskell Setup copy --destdir=${pkgdir}
	install -D -m644 LICENSE ${pkgdir}/usr/share/licenses/$pkgname/LICENSE
}
