$RO 78.109.128.1/32 $GT
$RO 78.109.128.2/32 $GT
$RO 172.16.20.58/32 $GT
# маска 255.255.254.0 соответственно, /23

for i in 20 28 30 32 34 36 40 42 44 46 48
do
$RO 172.16.$i.0/23 $GT
done
