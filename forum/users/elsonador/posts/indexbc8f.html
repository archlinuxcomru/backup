# $Id: PKGBUILD 55879 2011-09-24 16:48:26Z stephane $
#Maintainer: Stéphane Gaudreault <[email protected]>;
#Contributor: BlackEagle < ike DOT devolder AT gmail DOT com >
#Contributor: Dany Martineau <[email protected]>;
pkgname=clementine-reconf
pkgver=0.7.1
pkgrel=1
pkgdesc="A music player and library organizer"
url="http://www.clementine-player.org/"
license=('GPL')
arch=('i686' 'x86_64')
depends=('gstreamer0.10' 'taglib' 'glew' 'liblastfm' 'libgpod' 'libmtp' 'libplist' 'hicolor-icon-theme' 'qt' 'libimobiledevice')
makedepends=('cmake' 'boost')
optdepends=('gstreamer0.10-base-plugins: for more open formats'
            'gstreamer0.10-good-plugins: for use with "Good" plugin libraries'
            'gstreamer0.10-bad-plugins: for use with "Bad" plugin libraries'
            'gstreamer0.10-ugly-plugins: for use with "Ugly" plugin libraries')
source=(http://clementine-player.googlecode.com/files/clementine-${pkgver}.tar.gz
        clementine-0.7.1-fix-devicekit.patch)
sha1sums=('8b2025b8876f9f3fe33ef5001e0621f3a5d0f142'
          'd595e2746949363680e0d64fe2fb97b524c8d27c')
conflicts=('clementine')
install=clementine.install
build() {
   cd "${srcdir}/clementine-${pkgver}"
   patch -Np1 -i ../clementine-0.7.1-fix-devicekit.patch
   cmake . -DCMAKE_INSTALL_PREFIX=/usr -DCMAKE_BUILD_TYPE=Release -DSTATIC_SQLITE=0 -DENABLE_GIO=OFF -DENABLE_WIIMOTEDEV=OFF -DENABLE_LIBGPOD=OFF -DENABLE_IMOBILEDEVICE=OFF -DENABLE_LIBMTP=OFF -DENABLE_VISUALISATIONS=OFF -DBUNDLE_PROJECTM_PRESETS=OFF -DENABLE_REMOTE=OFF
   make
}
package() {
   cd "${srcdir}/clementine-${pkgver}"
   make DESTDIR="${pkgdir}" install
}
