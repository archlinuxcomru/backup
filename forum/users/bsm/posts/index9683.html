Проблема с автомонтированием USB-устройств (Flash, Disk)-

1. независимо от типа файловой системы USB-устройства {vfat,ntfs}, оно монтируется как устройство с не NTFS файловой системой;
2. при наличии linux файловой системы устройство не монтируется.


Файл правил /etc/udev/rules.d/10-usb.rules:
#
# Пт. 10 февр. 2012 16:07:21
#
KERNEL!="sd[b-z][0-9]" GOTO="automount_end"
# Добавление устройства
#
ACTION=="add" RUN+="/bin/mkdir -p /mnt/%E{ID_VENDOR}_%E{ID_MODEL}_%n"
#
ACTION=="add" ENV{mount_options}="rw,utf8,nodev,noexec,nosuid,sync,uid=100"
#
# Определяем тип файловой системы- можно обойтись без этого, т.к. Windows FS
# {NTFS, vfat} монтируются без проблем, но Linux FS не монтируется.
#
ACTION=="add" ENV{ID_FS_TYPE}=="ntfs" RUN+="/bin/mount -t ntfs-3g -o %E{mount_options} /dev/%k /mnt/%E{ID_VENDOR}_%E{ID_MODEL}_%n" GOTO="automount_end"
ACTION=="add" ENV{ID_FS_TYPE}!="ntfs" RUN+="/bin/mount -t %E{ID_FS_TYPE} -o %E{mount_options} /dev/%k /mnt/%E{ID_VENDOR}_%E{ID_MODEL}_%n" GOTO="automount_end"
# Удаление устройства
#
ACTION=="remove" RUN+="/bin/umount /mnt/%E{ID_VENDOR}_%E{ID_MODEL}_%n" RUN+="/bin/rmdir /mnt/%E{ID_VENDOR}_%E{ID_MODEL}_%n"
# Предоставим возможность автомонтирования всем пользователям.
#
%wheel ALL= NOPASSWD: /bin/umount /mnt/%E{ID_VENDOR}_%E{ID_MODEL}_%n"
LABEL="automount_end"

На данный момент выполняю монтирование/демонтирование CD/DVD привода скриптом-
#!/bin/sh
#
# Монтирование CD/DVD устройств
#
# Чт. 02 февр. 2012 08:31:10
#
CD_DVD_DEV=/dev/sr0
TOCHKA_MOUNT=/mnt/dvdwriter
close() {
    cd /mnt
    umount $TOCHKA_MOUNT
    eject
    rmdir $TOCHKA_MOUNT
    status
}  # close()
open() {
    [ -d $TOCHKA_MOUNT ] || mkdir --parents $TOCHKA_MOUNT
    [ -d $TOCHKA_MOUNT ] || exit 1
    mount -t auto $CD_DVD_DEV $TOCHKA_MOUNT
    status
}  # open()
status() {
    status_device=`df -h | grep $CD_DVD_DEV`
    [ "$status_device" == "" ] && exit
    echo
    echo $status_device
    echo
    inform_cd_dvd_dev=`blkid $CD_DVD_DEV`
    echo $inform_cd_dvd_dev
    echo
}  # status()
case "$1" in
    open|close|status)
	$1
	;;
    *)
	echo "Монтирование CD/DVD устройств."
	echo "Используйте $0 {open|close|status}"
	exit 1
esac
exit 0

pacman -S iproute2

Сформировал перечень установленных в системе пакетов-

comm -13 <(pacman -Qmq | sort) <(pacman -Qqe | sort) > list_packages

В перечне пакета нет. При повторной установке пакета, pacman сообщает о наличии установки пакета и выполняет его повторную установке.
