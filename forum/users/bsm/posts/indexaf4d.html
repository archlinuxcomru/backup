Тогда предлагаю проверить - rfkill list
Выполнение rfkill list дало пустую строку.
Как не странно, всё заработало, возможно в системе живёт барабашка. Для последующего использования создал скрипт:

#!/bin/sh
#

WIFI_DEV=`iwconfig | grep wlp | awk '{ print $1 }'`

JANA_ESSID=Jana
JANA_PASSWD=22082208

WPA_CONF=${PWD}/wpa_supplicant.conf

[ -f $WPA_CONF ] || wpa_passphrase $JANA_ESSID $JANA_PASSWD > $WPA_CONF	# создание файла конфиг.

# Процедуры -------------------------------------------------------------------
#
grand_power() {	# увеличение мощности устройства
	ip link set $WIFI_DEV dowm	             # отключение интерфейса

	macchanger -r $WIFI_DEV
	iw reg set BO				             # BO- Боливия, BZ- Белиз
	iwconfig $WIFI_DEV txpower 500mW
}  # grand_power()

up_wifi() {		# поднимаем wifi
	#
	# "поднятие" интерфейса
	# соединение с (.) доступа
	# получение ip-address
	# проверка доступа к интернет
	#
	ip link set $WIFI_DEV up && \
	wpa_supplicant -B -Dwext -i $WIFI_DEV -c $WPA_CONF && \
	dhcpcd $WIFI_DEV && \
	ping -c 12 mirror.yandex.ru
}  # up_wifi()

# Головной модуль -------------------------------------------------------------
#
#grand_power		# увеличение мощности устройства
up_wifi			# поднимаем wifi
