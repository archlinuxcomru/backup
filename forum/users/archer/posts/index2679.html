Да, еще /etc/ppp/ip-up/ip-up-route
#!/bin/bash
PATH="/sbin:/bin:/usr/sbin:/usr/bin"
GATEWAY=`route | grep ^default | awk '{print $2}'`
route del default
route add default dev ppp0
(
echo '#!/bin/bash'
echo 'PATH="/sbin:/bin:/usr/sbin:/usr/bin"'
echo 'route del default'
echo "route add default gateway $GATEWAY"
) > /etc/ppp/ip-down

[lac corbina]
lns = tp.internet.beeline.ru
redial = yes
redial timeout = 5
require chap = yes
require authentication = no
name = 1234567890
ppp debug = no
pppoptfile = /etc/ppp/options.xl2tpd
require pap = no
autodial = yes
/etc/ppp/options.l2tpd
lcp-echo-interval 10
lcp-echo-failure 2
name 1234567890
remotename tp
ipparam corbina
connect /bin/true
mru 1460
mtu 1460
nodeflate
noauth
persist
maxfail 0
nobsdcomp
noaccomp
nopcomp
defaultroute
