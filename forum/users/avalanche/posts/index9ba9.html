Для того, чтобы inkscape мог открывать cdr формат нужно установить uniconvertor и немного шаманства:

sudo pacman -S inkscape
yaourt -S uniconvertor

затем делаем так:
cd /usr/bin
ln uniconv uniconvertor

все теперь cdr файлы можно открывать прямо из inkscape

Проблема в том, что inkscape 0.46-11 почему-то ищет команду uniconv по имени uniconvertor

тема закрыта, проблема в очередной раз решилась установкой exo-mountopt из AUR
sudo pacman -Rd exo
yaourt -S exo-mountopt
sudo vi /usr/share/xfce4/mount.rules

mount.rules:
[vfat]
mount_options=shortname=mixed;uid=;utf8;umask=077;exec;flush
# mount_options=shortname=mixed;uid=;utf8;umask=077;exec;usefree
# Sometimes, adding a 'usefree' option is required due to bugs in the kernel.
# See: https://bugs.launchpad.net/ubuntu/+source/nautilus/+bug/133567
[udf]
mount_options=uid=;exec
[ntfs-3g]
mount_options=locale=;exec
[ntfs]
mount_options=umask=222;utf8;exec
fstype_override=ntfs-3g
[iso9660]
mount_options=uid=;utf8;exec
[hfs]
mount_options=uid=
