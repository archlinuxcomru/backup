Закиньте сюда хоть тот PKGBUILD, а как так, на пальцах-то? “Поставить рабочую ссылку в source=(''), исправить номер версии, md5=('') …”
PKGBUILD:
# Contributor: AndrГ© Fettouhi <[email protected]>;
pkgname=eclipse-photran
pkgver=40b5
pkgrel=1
pkgdesc="IDE for Fortran 77, 90, and 95 based on Eclipse and the CDT"
url="http://www.eclipse.org/photran/"
arch=('i686' 'x86_64')
license=('custom')
depends=("eclipse")
source=(http://download.eclipse.org/technology/photran/updates/photran$pkgver-master.zip)
noextract=("photran$pkgver-master.zip")
md5sums=('d21a841114fe3766691a3432137344cb')
build() {
  installdir=$startdir/pkg/usr/share/eclipse
  cd $startdir/src || return 1
  unzip photran$pkgver-master.zip || return 1
  mkdir -p $installdir || return 1
  cp -a $startdir/src/features $installdir || return 1
  cp -a $startdir/src/plugins $installdir || return 1
  cp -a $startdir/src/site.xml $installdir || return 1
}
