pkgname=mc-slavaz
pkgver=4.6.2
pkgrel=1
pkgdesc="midnight commander"
arch=(i686 x86_64)
url=""
license=('GPL')
depends=()
makedepends=()
conflicts=(mc)
source=(http://people.redhat-club.org/inf/mc-slavaz/srpm/mc-4.6.2-el5.rhc.9.20081121svn.src.rpm)
md5sums=('3b68c0687e69a713c67c1101c9d1bac0')
build() {
  cd $srcdir
  [ $NOEXTRACT -eq 0 ] && rpmextract.sh mc-4.6.2-el5.rhc.9.20081121svn.src.rpm && \
    tar xzf mc-20081121.tar.gz
  cd mc-20081121
  ./configure --prefix=/usr --enable-charset --with-edit --with-vfs --with-mcfs --with-samba --with-x \
        --with-subshell --enable-extcharset --enable-netcode --with-configdir=/etc --with-screen=slang
  make || return 1
  make DESTDIR="$pkgdir/" install || return 1
  mkdir -p $pkgdir/usr/etc
  ln -s /usr/share/mc $pkgdir/usr/etc/mc
}

если 2 канала на одном компе, нужно разделить таблицы маршрутов

ip route add 1.2.3.0/24 dev isib table sibroutes
ip route add 192.168.1.0/24 dev idom table domroutes

ip route add default via 1.2.3.1 table sibroutes
ip route add default via 192.168.1.1 table domroutes

ip rule add from 1.2.3.4 table sibroutes
ip rule add from 192.168.1.5 table domroutes
