[Desktop Entry]
Type=Application
Name=IPTV rus
Name[ru]=IPTV rus
Comment=Воспроизвести плейлист
Comment[ru]=Воспроизвести плейлист
Icon=mpv
#TryExec=mpv
Exec=sh -c 'uplist; mpv ~/.config/mpv/playlists/rus.m3u'
Terminal=false
Categories=Video;Network;
MimeType=
X-KDE-Protocols=ftp,http,https,mms,rtmp,rtsp,sftp,smb,srt
