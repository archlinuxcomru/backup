Он создается при установке acestream-engine

acestream-engine.install
post_install() {
  if ! getent passwd acestream &> /dev/null; then
    useradd -u 4035 -g video -m -s /usr/bin/nologin acestream
  fi
}

post_upgrade() {
  post_install
}

post_remove() {
  if getent passwd acestream &>/dev/null; then
    userdel acestream
  fi
}
