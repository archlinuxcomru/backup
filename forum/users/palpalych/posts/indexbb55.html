У меня тоже Lenovo

sudo yaourt -S rfkill
rfkill unblock all
sudo ifconfig wlan0 up
sudo iwlist wlan0 scanning
sudo mv /etc/wpa_supplicant.conf /etc/wpa_supplicant.conf.original
sudo su
wpa_passphrase имя_точки пароль > /etc/wpa_supplicant.conf
exit
sudo gedit /etc/wpa_supplicant.conf
ctrl_interface=DIR=/var/run/wpa_supplicant GROUP=wheel
network={
	ssid="имя_точки"
	psk=f3c4c71562accfbed9301... что-то типа этого...ce3675ef0ccbc8ebe9
}
sudo iwconfig wlan0 essid имя_точки
sudo wpa_supplicant -B -Dwext -i wlan0 -c /etc/wpa_supplicant.conf
sudo dhcpcd wlan0
потом поднимать в ручную:
sudo ifconfig wlan0 up
sudo dhcpcd wlan0
или автоматом:
/etc/rc.conf
....
eth0="dhcp"
wlan0="dhcp"
INTERFACES=(eth0 wlan0)
....
DAEMONS=(... !network ...)
....
/etc/rc.local
wpa_supplicant -B -Dwext -iwlan0 -c/etc/wpa_supplicant.conf
dhcpcd wlan0
ifconfig eth0 up
dhcpcd eth0
sudo reboot...
