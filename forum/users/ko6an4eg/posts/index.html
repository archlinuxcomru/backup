Листинг rc.conf

#
# /etc/rc.conf - Main Configuration for Arch Linux
#
# -----------------------------------------------------------------------
# LOCALIZATION
# -----------------------------------------------------------------------
#
# LOCALE: available languages can be listed with the 'locale -a' command
# HARDWARECLOCK: set to "UTC" or "localtime"
# USEDIRECTISA: use direct I/O requests instead of /dev/rtc for hwclock
# TIMEZONE: timezones are found in /usr/share/zoneinfo
# KEYMAP: keymaps are found in /usr/share/kbd/keymaps
# CONSOLEFONT: found in /usr/share/kbd/consolefonts (only needed for non-US)
# CONSOLEMAP: found in /usr/share/kbd/consoletrans
# USECOLOR: use ANSI color sequences in startup messages
#
LOCALE="en_US.utf8"
HARDWARECLOCK="localtime"
USEDIRECTISA="no"
TIMEZONE="Europe/Moscow"
KEYMAP="us"
CONSOLEFONT=
CONSOLEMAP=
USECOLOR="yes"
# -----------------------------------------------------------------------
# HARDWARE
# -----------------------------------------------------------------------
#
# MOD_AUTOLOAD: Allow autoloading of modules at boot and when needed
# MOD_BLACKLIST: Prevent udev from loading these modules
# MODULES: Modules to load at boot-up. Prefix with a ! to blacklist.
#
# NOTE: Use of 'MOD_BLACKLIST' is deprecated. Please use ! in the MODULES array.
#
MOD_AUTOLOAD="yes"
#MOD_BLACKLIST=() #deprecated
MODULES=()
# Scan for LVM volume groups at startup, required if you use LVM
USELVM="no"
# -----------------------------------------------------------------------
# NETWORKING
# -----------------------------------------------------------------------
#
# HOSTNAME: Hostname of machine. Should also be put in /etc/hosts
#
HOSTNAME="myhost"
# Use 'ifconfig -a' or 'ls /sys/class/net/' to see all available interfaces.
#
# Interfaces to start at boot-up (in this order)
# Declare each interface then list in INTERFACES
#   - prefix an entry in INTERFACES with a ! to disable it
#   - no hyphens in your interface names - Bash doesn't like it
# 
# DHCP:     Set your interface to "dhcp" (eth0="dhcp")
# Wireless: See network profiles below
#
#Static IP example
eth0="eth0 192.168.1.2 netmask 255.255.255.0 broadcast 192.168.1.255"
#eth0="dhcp"
INTERFACES=(eth0)
# Routes to start at boot-up (in this order)
# Declare each route then list in ROUTES
#   - prefix an entry in ROUTES with a ! to disable it
#
gateway="default gw 192.168.1.1"
ROUTES=(!gateway)
 
# Enable these network profiles at boot-up.  These are only useful
# if you happen to need multiple network configurations (ie, laptop users)
#   - set to 'menu' to present a menu during boot-up (dialog package required)
#   - prefix an entry with a ! to disable it
#
# Network profiles are found in /etc/network.d
#
# This now requires the netcfg package
#
#NETWORKS=(main)
# -----------------------------------------------------------------------
# DAEMONS
# -----------------------------------------------------------------------
#
# Daemons to start at boot-up (in this order)
#   - prefix a daemon with a ! to disable it
#   - prefix a daemon with a @ to start it up in the background
#
DAEMONS=(syslog-ng network netfs crond)

Добрый день. У меня следующая проблема: решил дома развернуть тестовый web-сервачок на базе ArchLinux 2009.8. Выход в интернет осуществляется через ADSL модем, настроенный роутером (все настройки подключения прописаны на нем), к нему подключены 3 компа (в том числе этот сервачок). Так что в rc.conf нужно указать только ip компа (устанавливал статический ip внутренней сети, закоментировав dhcp), шлюза (поставил адрес ADSL модема) и в resolv.conf адреса DNS серверов (адреса DNS серверов провайдера). После настроек попытался пропинговать адреса известных web-сайтов по-имени, например: ya.ru - пинга нет. После чего попытался пропинговать адреса DNS серверов - тоже не пингуются, однако внутренняя сеть пингуется спокойно, модем тоже пингуется. А на винде, установленной на том же компе - все работает корректно, настройки те же самые (сейчас пишу с того же компа, - альтернативно установлена винда). Такая же проблема была на рабочем компе, причем если комп получал ip по dhcp, то сеть начинала работать, но тоже с перебоями - почему-то сеть не всегда хотела работать, то пинги адресов web-сайтов работают, то нет… Решала перезагрузка компа, перезагрузка демона сети не помогала).

Материнка на домашнем компе: ASUS 5B Deluxe, встроенная сетевая карта домашнего компа: Marvell Yukon 88E8056 PCI-E Gigabit Ethernet Controller

Кто-нибудь встречался с подобной проблемой?
