Тоже забил на netcfg.
Написал небольшой скрипт для коннекта
#!/bin/bash
. /etc/rc.conf
. /etc/rc.d/functions
case "$1" in 
     start)
     stat_busy "Starting wifi"
     wpa_supplicant -Dwext -iwlan0 -c/etc/wpa_supplicant.conf -B
     screen -d -m dhcpcd wlan0 
     if [ $? -gt 0 ]; then        
       stat_fail                  
     else                         
       add_daemon rtorrent        
       stat_done                  
     fi
     ;;
     stop)
     stat_busy "Stopping wifi"
     killall dhcpcd
     killall wpa_supplicant
     #rm -f /var/run/wpa_supplicant/wlan0
     #rm -f /var/run/dhcpcd-wlan0.pid
     if [ $? -gt 0 ]; then
       stat_fail
     else                 
       rm_daemon rtorrent 
       stat_done          
     fi    
     ;;
     restart)
     $0 stop
     sleep 1
     wpa_supplicant -Dwext -iwlan0 -c /etc/wpa_supplicant.conf -B
     dhcpcd wlan0
     ;;
     *)
     echo "usage: $0 {start|stop|restart}"
esac
