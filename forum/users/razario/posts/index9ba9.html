Ну думаю соберу свой.. Скачал исходники (вот эту версию kbdd-0.6.2.tar.gz). Скачал для примера PKGBUILD ( этого и этого) и вдумчиво читал Wiki и прочие ссылки…

Создал каталог, где буду собирать… подправил PKGBUILD
# Maintainer: Bla_bla
pkgname=kbdd
pkgver=0.6.2
pkgrel=1
pkgdesc="Simple daemon and library to make per window layout."
arch=('i686' 'x86_64')
url="https://github.com/qnikst/kbdd"
license=('GPL3')
depends=()
source=("https://github.com/downloads/qnikst/$pkgname/$pkgname-$pkgver.tar.gz)
md5sums=('df2ad5cf9f9c723fd24fbf0577a2b4af')
build() {
    cd $startdir/src/$pkgname-$pkgver
    ./configure --prefix=/usr
    make || return 1
    make prefix=$startdir/pkg/usr install
}
