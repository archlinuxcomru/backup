...
mynetworks_style = host
mynetworks = 127.0.0.1, <ip1>, <ip2>, <ip3>, ... , <ext-ip>
...
# SASL parameters
smtpd_sasl_auth_enable = yes
smtpd_sasl_security_options = noanonymous
...
smtpd_sender_login_maps = mysql:/etc/postfix/mysql_sender_login_maps.cf
...
smtpd_client_restrictions =
    permit_mynetworks,
    check_client_access cidr:/etc/postfix/reject_unauthenticated,
    permit_sasl_authenticated,
    reject_rbl_client dnsbl.sorbs.net,
    reject_rbl_client rhsbl.sorbs.net,
    reject_rbl_client sbl.spamhaus.org,
    reject_rbl_client bl.spamcop.net,
    reject_rbl_client relays.mail-abuse.org
smtpd_helo_required = yes
smtpd_helo_restrictions =
    permit_mynetworks,
    permit_sasl_authenticated,
    reject_invalid_helo_hostname,
    reject_invalid_hostname,
    reject_non_fqdn_helo_hostname,
    reject_unknown_helo_hostname
smtpd_sender_restrictions =
    permit_mynetworks,
    reject_sender_login_mismatch,
    permit_sasl_authenticated,
    reject_non_fqdn_sender,
    reject_unknown_sender_domain
smtpd_recipient_restrictions =
    permit_mynetworks,
    permit_sasl_authenticated,
    check_policy_service inet:127.0.0.1:60000,
    reject_unknown_recipient_domain,
    reject_unauth_destination
smtpd_data_restrictions =
    reject_unauth_pipelining
2. # cat /etc/postfix/reject_unauthenticated
192.168.0.0/24 permit_sasl_authenticated, reject
при такой конфигурации мы имеем:
1. отправка писем с нашего сервера в любом направлении только прошедшим авторизацию + без авторизации с указанных ip (см. mynetworks)
2. при отправке письма из ЛС на наш же сервер авторизация так же обязательна (см. check_client_access)
3. при отправке письма снаружи на наш сервер (с другого сервера) обязательно прохождение след. проверок:
. 3а. rbl
. 3b. greylist
. 3с. amavisd (это отдельная статья расходов :) )
4. проверка соответствия логин/ящик (см. reject_sender_login_mismatch)

иными? словами:
отправка с нашего сервера доступна только авторизированым пользователям, независимо от того, где они находятся (внутри ЛС или снаружи), + исключения mynetworks (по ip);
запрещено отправлять письма от чужого имени + исключения mynetworks;
неавторизированым пользователям/серверам можно отправлять письма на наш сервер только снаружи и только пройдя ряд проверок (rbl, greylist)

думаю/надеюсь, одно из двух объяснений станет понятно :)
