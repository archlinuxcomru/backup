Почитал я это и в итоге сварганил такое:
~/.bin/translate-en-ru
#!/bin/sh

text1="$(xclip -o | sed "s/[\"\'<>]//g")"
text2="en > ru:"

translate="$(xclip -o | trans \
-show-translation-phonetics n \
-show-original n \
-show-languages n \
-show-prompt-message n \
-show-dictionary n \
-show-alternatives n \
-no-ansi \
en:ru)"

notify-send -u low "$text2" "$translate"
