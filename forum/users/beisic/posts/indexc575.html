если нажму пробел на русском языке, переключу на английскйи, введу пакман, то будет вместо pacman вот эта брехня введена 321\204\320pacman
Если, установку системы, начинали с # loadkeys ru , то вся система русифицирована :
 beisic ~ $  cat /etc/locale.conf
LANG=ru_RU.UTF-8
Или, не сделали настройку файла ~.bashrc, например :
 beisic ~ $  cat .bashrc
#
# ~/.bashrc
#

# If not running interactively, don't do anything
[[ $- != *i* ]] && return

export HISTCONTROL=ignoreboth
export HISTIGNORE='history*'
export PROMPT_COMMAND='history -a;echo -en "\e]2;";history 1|sed "s/^[ \t]*[0-9]\{1,\}  //g";echo -en "\e\\";'
alias ls='ls --color=auto'
alias merge="xrdb -merge ~/.Xresources"
alias pacman='sudo pacman --color auto'
alias update='sudo pacman -Syu'
alias ps="ps auxf"
alias psgrep="ps aux | grep -v grep | grep -i -e VSZ -e"
alias fixpng="find . -type f -name "*.png" -exec convert {} -strip {} \;"
alias fc='sudo fc-cache -fv'
alias mirror="sudo reflector --protocol https --latest 50 --number 20 --sort rate --save /etc/pacman.d/mirrorlist"
PS1='\[\e[0;32m\]\u\[\e[m\] \[\e[1;34m\]\w\[\e[m\] \[\e[1;32m\]\$\[\e[m\] \[\e[1;37m\]'
#shopt
shopt -s autocd
shopt -s cdspell
shopt -s cmdhist
