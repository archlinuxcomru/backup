sudo systemctl stop pulseaudio.socket
sudo systemctl stop pulseaudio.service
sudo killall pulseaudio
# дальше без судо от пользователя
systemctl --user stop pulseaudio.socket
systemctl --user stop pulseaudio.service
rm /home/slider/.config/pavucontrol.ini
rm -r /home/slider/.config/pulse
rm /home/slider/.pulse/default.pa
 rm  /home/slider/.cache/sessions/*

если ругнется что нет прав удалить удаляйте от рута, но пульсу от рута запускать не надо.
Дальше ставьте пакеты

sudo pacman -Syyu
sudo pacman -S pavucontrol pulseaudio pulseaudio-alsa alsa-card-profiles
Проверьте что нет пульсы в автостарте если есть удалите

cat  /home/slider/.config/autostart

Перезагрузитесь
Проверьте состояние сервиса

systemctl --user status pulseaudio.service
systemctl --user status pulseaudio.socket
systemctl --user start pulseaudio.socket
systemctl --user start pulseaudio.service
Если сервис и сокет без ошибок запускайте pavucontrol ( от юзера только) и настраивайте вывод звука.
pavucontrol
