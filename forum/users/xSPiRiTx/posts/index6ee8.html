/etc/X11/xorg.conf
Section "InputDevice"
		Identifier  "Keyboard0"
		Driver	  "keyboard"
		Option	  "CoreKeyboard"
		Option	  "XkbRules" "xorg"
		Option	  "XkbModel" "pc105"
		Option	  "XkbLayout" "ru,us"
		Option	  "XkbOptions" "grp:alt_shift_toggle,lv3:ralt_switch,grp_led:scroll"
EndSection

LOCALE="ru_RU.UTF-8"
HARDWARECLOCK="local"
