Можно вот такой скриптик заюзать
#!/bin/bash
touchpad_status=`synclient | grep TouchpadOff | awk '{print $3}'`
if [ $touchpad_status == "0" ];
        then
                synclient TouchpadOff=1
                echo "TouchpadDisabled"
        else
                synclient TouchpadOff=0
                echo "TouchpadEnabled"
fi
