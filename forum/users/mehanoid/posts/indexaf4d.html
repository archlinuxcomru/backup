Пример использования в комментарии есть =)
#usage: mynotify-send 1appname 2icon 3title 4message 5timeout
#example: mynotify-send mpd media-playback-pause "Музыкальный плеер" "Пауза" 5000
appname - чтобы сообщение появлялось только один раз, в качестве appname нужно указывать одно и то же значение, т.е. следующие сообщения с таким же appname будут заменять предыдущее.
icon - название системной иконки, с которой будет показываться уведомление (можно пустую строку указать)
title - заголовок сообщения
message - само сообщение
timeout - время показа в миллисекундах

Сделать хотя бы так, чтобы сообщение обновлялось, а не новое приходило.

У меня для этой цели есть скриптик. Честно, не помню, где его взял, это было давно и не правда, может даже сам сваял))
Под кедами работает, в других DE вроде тоже, насколько я помню.

#!/bin/bash
#usage: mynotify-send 1appname 2icon 3title 4message 5timeout
#example: mynotify-send mpd media-playback-pause "Музыкальный плеер" "Пауза" 5000
notifyid=0
filedir=/tmp/mynotify
filename="$filedir""/$1"
[[ -e $filename ]] && read notifyid < $filename || [[ -e $filedir ]] || mkdir $filedir
notifyid=$(gdbus call --session --dest org.freedesktop.Notifications --object-path /org/freedesktop/Notifications --method org.freedesktop.Notifications.Notify "\"$1\"" $notifyid "\"$2\"" "\"$3\"" "$4" \[\] {} "$5" | awk '{print $2;}' | sed s/,\)//)
echo $notifyid > $filename
