	local v_sep_constraint = common.constraint({height = beautiful.panel_padding_bottom, width = beautiful.left_panel_width})

	local v_sep = wibox.container.background(v_sep_constraint, beautiful.bg_normal)
	setmetatable(v_sep, { __index = v_sep_constraint })

-- Create the vertical wibox

	s.my_left_wibox = awful.wibar({
		position = "left",
		screen = s,
		width = beautiful.left_panel_width,
		bg = "#40445050",
		})

    s.my_left_wibox:setup {
		layout = wibox.layout.align.horizontal,

Полез в настройки видио

/etc/X11/xorg.conf/20-intel.conf

Section "Monitor"
	Identifier	"Monitor0"
EndSection

Section "Device"
	Identifier	"Device0"
	Driver		"intel" #Choose the driver used fo r this monitor
EndSection

Section "Screen"
	Identifier	"Screen0" #Collapse Monitor and Device section to Screen section
	Device		"Device0"
	Monitor		"Monitor0"
	DefaultDepth	24 #Choose the depth (16||24)
SubSection		"Display "
	Depth		24
	Modes		"1280x768_60.00" #Choose the resolution
EndSubSection
EndSection
