Там, в конфиге, столько лишнего мусора...
Фактически и так работает.. Это только начало конфига, дальше собственно выбор операционки

insmod gpt
insmod pc
insmod gfxmenu
insmod videotest
insmod tga
insmod png
insmod gfxterm
insmod lspci
insmod vbeinfo
insmod vbe
insmod ntfs
insmod chain
insmod biosdisk
insmod font

set default=0
set timeout=30
set fallback=1
search --no-floppy --fs-uuid --set=root 894a
search --no-floppy --fs-uuid --set=boot 23c6
set root=${root}
loadfont ($boot)/grub/fonts/unicode.pf2
loadfont ($boot)/grub/themes/SteamBP/DejaVuSans-20.pf2
set gfxmode=1920x1080
if terminal_output gfxterm ; then true ; else
terminal gfxterm
fi
set theme=($boot)/grub/themes/SteamBP/theme.txt
export theme
#####
1,2 загрузка шрифта
3 разрешение экрана
4 собствено тема

loadfont ($boot)/grub/themes/SteamBP/DejaVuSans-20.pf2
