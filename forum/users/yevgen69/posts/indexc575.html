fstab:
# 
# /etc/fstab: static file system information

tmpfs       /var/run    tmpfs   defaults,noatime,nodev,nosuid   0       0
tmpfs       /var/log    tmpfs   defaults,noatime,nodev,nosuid   0       0
/dev/sda1 /boot ext2 defaults 0 1
/dev/sda2 / ext4 defaults,noatime,nodiratime,discard,errors=remount-ro 0 1
/dev/sdb1 /home ext4 defaults 0 1
/dev/sdb2 swap swap defaults 0 0
