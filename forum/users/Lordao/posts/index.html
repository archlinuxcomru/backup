Это у меня в /etc/grub.d/40_custom. Зарезервированный раздел Windows находится на первом разделе.
#!/bin/sh
exec tail -n +3 $0
# This file provides an easy way to add custom menu entries.  Simply type the
# menu entries you want to add after this comment.  Be careful not to change
# the 'exec tail' line above.

menuentry 'Windows 10' -class windows -class os $menuentry_id_option  'osprober-chain-3E20CC2720CB$
savedefault
insmod part_msdos
insmod ntfs
set root='hd0,msdos1'
search --no-floppy --fs-uuid --set=root FC7C59207C58D6CC
ntldr /bootmgr
}
