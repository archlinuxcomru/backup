LOCALE="ru_RU.UTF-8"
HARDWARECLOCK="localtime"
USEDIRECTISA="no"
TIMEZONE="Europa/Moscow"
KEYMAP="ruwin_ct_sh-UTF-8"
CONSOLEFONT="/usr/share/kbd/consolefonts/cyr-sun16.psfu.gz"
CONSOLEMAP="cp866_to_uni.trans"
USECOLOR="yes"

LANG=C
