#!/usr/bin/perl
my $n = (`amixer -i | grep 'Mono: Playback 87' | cut -c 38-39`);
chomp ($n);
print "$n\n";
if ($n == "on")
{
    amixer -q set Master mute
}
else
{
     amixer -q set Master unmute
}
