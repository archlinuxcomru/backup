В debian было просто - создаешь файл restart.desktop с содержимым:
#!/bin/bash
[Desktop Entry]
Name=РЕстарт нетворк
Exec=gksu > lxterminal /etc/init.d/networking restart
Type=Application
Terminal=true
Icon=/home/l/Изображения/обои/network-restart.png
Name[ru_RU]=Рестарт
