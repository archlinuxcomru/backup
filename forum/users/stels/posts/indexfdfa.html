Помогите разобраться со звуком. Хочу, чтобы при проигрывании стерео фала играли все 6 колонок. А при просмотре двд звук правильно распределялся по колонкам. И еще, чтобы можно было слышать звук сразу изнескольких источников, например mplayer + amarok. Вот что имею:
.asoundrc
pcm.nforce-hw {
type hw
card 0
}
pcm.dsp0 {
    type plug
    slave.pcm "hw:0"
}
pcm.dmix5 {
    type dmix
    ipc_key 1024
    ipc_key_add_uid false
    ipc_perm 0660
    slave {
        pcm "hw:0"
        period_time 0
        period_size 1024
        buffer_time 0
        buffer_size 4096
        rate 48000
        channels 6
        }
}
pcm.dupl {
        type plug
        slave.pcm "dmix5"
        #slave.pcm "surround51"
        slave.channels 6
        route_policy duplicate
}
ctl.mixer0 {
type hw
card 0
}
