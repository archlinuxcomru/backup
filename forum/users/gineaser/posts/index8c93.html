 http_port 3128 intercept
И завернуть на iptables правилом 
iptables -t nat -A PREROUTING -i eth0 -p tcp --dport 80 -j REDIRECT --to-port 3128

ACTION=="add" RUN+="/bin/mkdir -p /mnt/%E{ID_VENDOR}_%E{ID_MODEL}_%n"
ACTION=="add" ENV{ID_FS_TYPE}=="ntfs" ENV{ID_FS_TYPE}!="ntfs" RUN+="/bin/mount -t ntfs-3g -o defaults /dev/%k  /mnt/%E{ID_VENDOR}_%E{ID_MODEL}_%n"
GOTO="automount_end"
ACTION=="add" RUN+="/bin/mount -t `/sbin/blkid -o value | tail -1` -o defaults /dev/%k /mnt/%E{ID_VENDOR}_%E{ID_MODEL}_%n"
GOTO="automount_end"
ACTION=="remove" RUN+="/bin/umount /mnt/%E{ID_VENDOR}_%E{ID_MODEL}_%n" RUN+="/bin/rmdir /mnt/%E{ID_VENDOR}_%E{ID_MODEL}_%n"
%wheel ALL= NOPASSWD: /bin/umount /mnt/%E{ID_VENDOR}_%E{ID_MODEL}_%n"
