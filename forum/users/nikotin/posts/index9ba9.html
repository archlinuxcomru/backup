Поиск в репах и в ауре:
#!/bin/sh
syncdump=`mktemp 2>/dev/null` || syncdump=/tmp/$$_syncdump
instpkg=`mktemp 2>/dev/null` || instpkg=/tmp/$$_instpkg
trap "rm -f $instpkg $syncdump" $?
from_repos() {
	pacman -Qsi $1 | egrep '^[^ ]' | sed '[email protected]^local/\(.*\s\)[][]\(.*\)[email protected]\[email protected]' > $instpkg
	pacman -Ss $1 > $syncdump
}
from_aur() {
	wget -q -O - "http://aur.archlinux.org/rpc.php?type=search&arg=$1" |
	sed '[email protected]{"ID":@\[email protected]' | sed '1d'| grep -Fv '"URLPath":""' |
	awk '{sortkey=$0; sub(/^.*[{,]"Name":"/, "", sortkey); sub(/"[,}].*$/, "", sortkey); print sortkey $0}' |
	sort | awk '{sub(/"[^"]+"/, ""); print $0}' | while read jsoninfo; do
		package=$(parse_json Name)
		version=$(parse_json Version)
		description=$(parse_json Description)
		numvotes=$(parse_json NumVotes)
		outofdate=$(parse_json OutOfDate)
		if [ $outofdate -eq 1 ]; then
			line="aur/${package} ${version} [votes: $numvotes, \\033[7;36m\\033[1;31mПакет устарел\\033[0;36m\\033[1;36m]"
		else
			line="aur/${package} ${version} [votes: $numvotes]"
		fi
		echo $line >> $syncdump
		echo "    $description" | sed 's/\([^ ].\{70\}\)/\1\n    /g' >> $syncdump
	done
}
parse_json() {
	echo $jsoninfo | sed -e 's/^.*[{,]"'$1'":"//' -e 's/"[,}].*$//'
}
mark_local() {
	sed -i '[email protected]^[^ ]@\n&@;[email protected]^[ ]@ \\033[1;30m&@' $syncdump
	sed -i '[email protected]^core/.*@\\033[1;31m[&]@g;[email protected]^extra/.*@\\033[1;32m[&]@g;[email protected]^community/.*@\\033[1;34m[&]@g;[email protected]^testing/.*@\\033[1;35m[&]@g;[email protected]^aur/.*@\\033[1;36m[&]@g;[email protected]^[^\ ].*/.*@\\033[1;33m[&]@g' $syncdump
	cat $instpkg | while read line; do
		pkgname=$(echo $line | sed "[email protected]\(^.*\) .*@\[email protected]")
		pkgver=$(echo $line | sed "[email protected]^.* \(.*\)@\[email protected]")
		sed -i "[email protected]^\([^ ].*\/$pkgname \)@\\\\033[5;32m↓ \\\\033[0;37m\[email protected];" $syncdump
		sed -i "[email protected]^.*↓ \(.*\/$pkgname $pkgver\)@\\\\033[1;29m↓ \[email protected];" $syncdump
	done
}
show_result() {
	echo -e "`< $syncdump`" 2>/dev/null | less
}
from_repos $1
from_aur $1
mark_local
show_result
exit 0

У меня вот некоторое время спустя, после запуска и после того как я запущу несколько приложений, весь интерфейс начинает тупить. Тупит все, начиная от открытия меню, скролл и выделение файлов/папок в dolphin и т.д. При этом такие эффекты kwin, как “все окна” или “все рабочие столы” работают плавно и гладко. Пробовал менять оформление — не помогло.

Понять кто виноват я не могу. Либо это иксы, либо дрова nvidia, а может это из-за qt, хз. Но, надо заметить, GTK приложения тупят так же, скролл в фф становится медленным и хреновым. Если несколько раз прокрутить колесо мыши то еще несколько секунд надо ждать пока прокручивается страница (а сразу после запуска скроллинг работает очень гладко и быстро). Так что скорей всего тулкит тут не причем, хотя и с уверенностью не скажешь тоже.

Хочу заметить что если открыть только dolphin и к примеру firefox то тормоза могут и не проявится вообще. Но с течением времени и с запуском все большего количества приложений они проявляются все сильней и сильней. В итоге когда у меня запущен amarok, kopete, kwallet, korganizer, ktorrent, kmail, dolphin, firefox, vlc, yakuake то тормоза становятся очень напрягающими. Коллекция в amarok прокручивается с невероятными тормозами. При этом если прекратить все действия со своей стороны, то загрузка процессора 5-10% (в основном amarok и ktorrent). Но обыкновенное перемещение мыши над виджетами грузит процессор почти на всю катушку.

Меня это все жутко бесит и вот я не могу понять, или это у меня что то очень специфическое или это все таки не так. Арч установлен уже несколько лет, когда все эти тормоза начались я и не заметил, но уже довольно давно, на 4.3 уже были и на 4.2 вроде тоже, а может и еще раньше. Система AMD x86_64, видео nvidia 9500GT, перед ней была 7600GT. Может пора пойти вендовым путем и переустановить арч, может что то запоролось, о чем я не знаю? Или это КДЕ такое г*вн*? Или X? Или nvidia? Или надо что то почистить? Я уже хз чего думать :)
