[trinity] Server = http://repo.nasutek.com/arch/contrib/trinity/x86_64/
Пакеты этого репозитория работают на Arch не старше 08.2018, т.е. нужен Downgrade системы
пишем зеркало в /etc/pacman.d/mirrorlist и откатим все пакеты: pacman -Syyuu
# Server = http://archlinux.arkena.net/archive/repos/2018/08/01/$repo/os/$arch
Преимущества: на таком Trinity работают все темы оформлений: Plastic QTCurve и прочие…

Второй способ: просто добавить репозиторий Condres, получим плюшки Manjaro и TDE 14.0.5, в котором работают только старые темы.
[condresrepo]
Server = http://build.codelinsoft.it/mirror/

Поставил тринити … - не работает переключалка раскладок - не реагирует на комбинации клавиш, только по щелчку мыши переключает
Возможно, в Trinity R14.0.5 это решено, но для R14.0.4 нужно настроить переключение раскладки в /etc/X11/xorg.conf.d/00-keyboard.conf (или применить параметры setxkbmap в автозапуске) и использовать для отображения языка ввода стороннюю утилиту: qxkb, sbxkb, xxkb…
# systemd-localed and Xorg. Use localectl(1) to instruct systemd-localed to update it.

Section "InputClass"
        Identifier "system-keyboard"
        MatchIsKeyboard   "on"
        Option "XkbModel" "pc105"
        Option "XkbLayout" "us+typo,ru:2+typo"
        Option "XkbOptions" "grp:caps_switch,grp:rctrl_switch,grp:lctrl_lshift_toggle,grp:lalt_lshift_toggle,grp:shift_caps_switch,shift:both_capslock_cancel,misc:typo,compose:rwin,lv3:ralt_switch,keypad:pointerkeys,grp_led:scroll,terminate:ctrl_alt_bksp"
EndSection

# Преимущества данного набора ключей: смена раскладки по [Left] Alt+Shift Ctrl+Shift рус: Alt+Caps англ: Caps. LShift+RShift = CapsLock. RCtrl инверсия языка ввода. RAlt: Level3 при смене раскладки не меняется, RWin: Compose. Shift+Num управление мышью с цифрового поля

# параметры работают в /etc/X11/xorg.conf.d/00-keyboard.conf, но при подключении через setxkbmap ошибка Error loading new keyboard description
# setxkbmap -layout us+typo,ru:2+typo -option lv3:ralt_switch,keypad:pointerkeys

# упрощенный режим, ошибок setxkbmap нет:
# setxkbmap -layout us,ru -option grp:caps_switch,grp:rctrl_switch,grp:lctrl_lshift_toggle,grp:lalt_lshift_toggle,grp:shift_caps_switch,shift:both_capslock_cancel,misc:typo,compose:rwin,grp_led:scroll,terminate:ctrl_alt_bksp
