вот мой конфиг, как-то тоже был мучался не мог расшарить, с этим работает, win7-8 пускает
#======================= Global Settings =====================================
[global]
   workgroup = WORKGROUP
   server string = Arch Linux Samba Server
   netbios name = Dell-1700
   security = user
   usershare path = /var/lib/samba/usershares
   usershare allow guests = yes
   usershare max shares = 100
   usershare owner only = no
   map to guest = Bad User

   printcap name = /etc/printcap
   load printers = yes

   log file = /var/log/samba/%m.log
   max log size = 50

   dns proxy = no

#============================ Share Definitions ==============================
[printers]
   comment = All Printers
   path = /var/spool/samba
   browseable = no
# Set public = yes to allow user 'guest account' to print
   guest ok = yes
   writable = no
   printable = yes

конфиг шары в /var/lib/samba/usershares/ForAll
#VERSION 2
path=/media/1Tb/ForAll
comment=
usershare_acl=S-1-1-0:R
guest_ok=y
sharename=ForAll
