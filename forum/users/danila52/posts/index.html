pkgname=pdftk
pkgver=1.41
pkgrel=3
pkgdesc="Pdftk is a simple tool for doing everyday things with PDF documents. Binary from Debian lenny"
url="http://www.pdfhacks.com/pdftk"
license="GPL"
arch=('i686' 'x86_64')
depends=('gcc-gcj')
source=("http://www.pdfhacks.com/${pkgname}/${pkgname}-${pkgver}.tar.gz"
"debian-build_with_gcc-4.3.diff"
"debian-no_embedded_libs.diff")
options=('!makeflags')
md5sums=('d8057a3ae8c3af7dfc7ea110f481ad7a'
         '033e054a2c4df9b11592ef393503dc75'
         'f12194c0a17afd029f3f26f0216b8ea6')
build() {
    cd $startdir/src/$pkgname-$pkgver
    patch -Np1 -i $startdir/src/debian-build_with_gcc-4.3.diff || return 1
    patch -Np1 -i $startdir/src/debian-no_embedded_libs.diff || return 1
    cd $startdir/src/$pkgname-$pkgver/$pkgname
    sed 's:-O2:\$(CFLAGS):g' -i Makefile.Generic
    unset CLASSPATH
    unset JAVA_HOME
    make -f Makefile.Generic || return 1
    install -D -m 755 pdftk $startdir/pkg/usr/bin/pdftk
}
